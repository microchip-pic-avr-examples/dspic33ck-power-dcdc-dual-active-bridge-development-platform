53void Dev_PwrCtrl_Initialize(void)
54{
55 dab.Properties.VPriReference = 0;
56 dab.Properties.VSecReference = 0;
57 dab.Properties.IReference = 0;
58 dab.Properties.PwrReference = 0;
59
60 // Initialize the PWM instance that the user initialize in MCC
61 // for the power control driver
62 dab.Pwm.Primary_1 = PWM_GENERATOR_1;
63 dab.Pwm.Primary_2 = PWM_GENERATOR_3;
64 dab.Pwm.Secondary_1 = PWM_GENERATOR_2;
65 dab.Pwm.Secondary_2 = PWM_GENERATOR_4;
66
67 // Initialize the Control Period and Control Phase during start-up
68 dab.Pwm.ControlPeriod = MAX_PWM_PERIOD;
69 dab.Pwm.ControlPhase = 0;
70 dab.Pwm.DeadTimeHigh = PG1DTH; // ToDo: remove the register dependency
71 dab.Pwm.DeadTimeLow = PG1DTL; // ToDo: remove the register dependency
72 dab.Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
73 dab.Pwm.PBVControlPhaseTarget = 0;
74
75 // Initialize the DAB to charging state
76 dab.PowerDirection = PWR_CTRL_CHARGING;
77
78 // Initialize Start-Up ramp settings
79 Dev_PwrCtrl_StartUpInitialize();
80
81 // Initialize Power Control Loop
82 Dev_PwrCtrl_ControlLoopInitialize();
83
84 // Disable PWM output by setting the PWM override bits to High
85 Dev_PwrCtrl_PWM_Disable(&dab);
86
87 // Enable PWM peripheral
88 PWM_Enable();
89
90 // Update the Period, Duty Cycle and Phases of the PWMs based on
91 // the given Control period and Control Phase
92 Dev_PwrCtrl_PWM_Update(&dab);
93}

163static void Dev_PwrCtrl_StartUpInitialize(void)
164{
165 // Initialize Voltage ramp-up settings
166 dab.VRamp.ptrReference = (uint16_t*)&dab.VLoop.Reference;
167 dab.VRamp.ptrReferenceTarget = &dab.Properties.VSecReference;
168 dab.VRamp.StepSize = 1;
169 dab.VRamp.Delay = 0;
170 dab.VRamp.Counter = 0;
171 dab.VRamp.RampComplete = 0;
172
173 //Initialize Current ramp-up settings
174 dab.IRamp.ptrReference = (uint16_t*)&dab.ILoop.Reference;
175 dab.IRamp.ptrReferenceTarget = (uint16_t*)&dab.Properties.IReference;
176 dab.IRamp.StepSize = 1;
177 dab.IRamp.Delay = 0;
178 dab.IRamp.Counter = 0;
179 dab.IRamp.RampComplete = 0;
180
181 //Initialize Power ramp-up settings
182 dab.PRamp.ptrReference = (uint16_t*)&dab.PLoop.Reference;
183 dab.PRamp.ptrReferenceTarget = &dab.Properties.PwrReference;
184 dab.PRamp.StepSize = 1;
185 dab.PRamp.Delay = 0;
186 dab.PRamp.Counter = 0;
187 dab.PRamp.RampComplete = 0;
188
189#if (OPEN_LOOP_PBV)
190 // Initialize Voltage ramp-up settings for Period control
191 dab.VRamp.ptrReference = &dab.Pwm.ControlPeriod;
192 dab.VRamp.ptrReferenceTarget = &dab.Pwm.PBVPeriodTarget;
193
194 //Initialize Current ramp-up settings for Phase control
195 dab.IRamp.ptrReference = &dab.Pwm.ControlPhase;
196 dab.IRamp.ptrReferenceTarget = &dab.Pwm.PBVControlPhaseTarget;
197#endif
198}
