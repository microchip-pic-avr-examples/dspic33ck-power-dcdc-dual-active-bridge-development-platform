209uint8_t PBV_UART_Receive_from_GUI()
210{
211 static uint16_t rcv_data_index = 0;
212 static uint16_t rcv_CRC = 0;
213 static uint16_t rcv_timeout = 0;
214 static uint8_t rcv_copy_for_CRC[PBV_RCV_DATABUFFER_SIZE + PBV_HEADER_SIZE];
215 uint8_t data;
216
217 while (1)
218 {
219 if (PBV_IsRxReady() == false)
220 {
221 if (++rcv_timeout >= 10000)
222 {
223 rcv_timeout = 0;
224 pbvUartObjectRx.UART_Frame_State = RCV_WAIT_FOR_STARTBYTE;
225 }
226 return PBV_MESSAGE_READY_TO_RECEIVE;
227 }
228 // ok, we have some data coming in, lets read and process
229 rcv_timeout = 0;
230 data = PBV_Read();
231 switch (pbvUartObjectRx.UART_Frame_State)
232 {
233 case RCV_WAIT_FOR_STARTBYTE:
234 if (data == PBV_START_OF_FRAME)
235 {
236 pbvUartObjectRx.UART_Frame_State = RCV_READ_ID_HIGHBYTE;
237 pbvUartObjectRx.Offset = 0;
238 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.Offset] = data;
239 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
240 rcv_data_index++; // for CRC
241 pbvUartObjectRx.Offset++;
242 }
243 return PBV_STATE_RECEIVING;
244
245 case RCV_READ_ID_HIGHBYTE:
246 pbvUartObjectRx.Protocol_ID = data << 8;
247 pbvUartObjectRx.UART_Frame_State = RCV_READ_ID_LOWBYTE;
248 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.Offset] = data;
249 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
250 rcv_data_index++; // for CRC
251 pbvUartObjectRx.Offset++;
252 return PBV_STATE_RECEIVING;
253
254 case RCV_READ_ID_LOWBYTE:
255 pbvUartObjectRx.Protocol_ID |= data;
256 pbvUartObjectRx.UART_Frame_State = RCV_READ_LENGTH_HIGHBYTE;
257 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.Offset] = data;
258 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
259 rcv_data_index++; // for CRC
260 pbvUartObjectRx.Offset++;
261 return PBV_STATE_RECEIVING;
262
263 case RCV_READ_LENGTH_HIGHBYTE:
264 // clear CRC flag, for now, as the firmware doesn't support it
265 // With PBV version 2.0.0, dsPIC cannot receive
266 // messages from the GUI if this flag is not cleared
267 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.Offset] = data;
268 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
269 rcv_data_index++; // for CRC
270 data = data & 0x7F; // clear bit 7 (CRC flag)
271 pbvUartObjectRx.Length_in_Bytes = data << 8;
272 pbvUartObjectRx.UART_Frame_State = RCV_READ_LENGTH_LOWBYTE;
273 pbvUartObjectRx.Offset++;
274 return PBV_STATE_RECEIVING;
275
276 case RCV_READ_LENGTH_LOWBYTE:
277 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.Offset] = data;
278 pbvUartObjectRx.Length_in_Bytes |= data;
279 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
280 rcv_data_index++; // for CRC
281 pbvUartObjectRx.UART_Frame_State = RCV_READ_DATA;
282 pbvUartObjectRx.Offset = 0;
283 return PBV_STATE_RECEIVING;
284
285 case RCV_READ_DATA:
286 if (pbvUartObjectRx.Offset < PBV_RCV_DATABUFFER_SIZE + PBV_HEADER_SIZE ) // MAX
287 {
288 pbvUartObjectRx.data[pbvUartObjectRx.Offset] = data;
289 pbvUartObjectRx.Offset++;
290
291 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
292 rcv_data_index++; // for CRC
293 }
294 if (pbvUartObjectRx.Offset >= pbvUartObjectRx.Length_in_Bytes) //are we finished receiving data???
295 {
296 pbvUartObjectRx.UART_Frame_State = RCV_READ_CRC_HIGHBYTE;
297 }
298 return PBV_STATE_RECEIVING;
299
300 case RCV_READ_CRC_HIGHBYTE:
301 rcv_CRC = data << 8;
302 pbvUartObjectRx.UART_Frame_State = RCV_READ_CRC_LOWBYTE;
303 return PBV_STATE_RECEIVING;
304
305 case RCV_READ_CRC_LOWBYTE:
306 rcv_CRC |= data;
307 pbvUartObjectRx.UART_Frame_State = RCV_READ_EOF;
308 return PBV_STATE_RECEIVING;
309
310 case RCV_READ_EOF:
311 if (data == 0x0d)
312 {
314 PBV_calculated_CRC = PBV_Calculate_CRC(rcv_copy_for_CRC, pbvUartObjectRx.Length_in_Bytes + PBV_HEADER_SIZE);
315 if (PBV_calculated_CRC == rcv_CRC)
316 {
317 rcv_data_index = 0;
318 rcv_CRC = 0;
319 PBV_calculated_CRC = 0;
320 pbvUartObjectRx.UART_Frame_State = RCV_MESSAGE_RECEIVED;
321 return PBV_MESSAGE_RECEIVED;
322 }
323 else {
324 rcv_data_index = 0; //for CRC
325 return PBV_MESSAGE_RX_ERROR;
326 }
327 }
328 }
329 }
330}

338uint8_t PBV_UART_Transmit_Ascii_to_GUI()
339{
340 uint8_t retval = PBV_MESSAGE_TX_ERROR;
341 uint8_t temp;
342 if (uartActiveTx == true)
343 {
344 retval = PBV_MESSAGE_TX_ERROR;
345 }
346 else
347 {
348 switch (pbvUartObjectAscii.UART_Frame_State)
349 {
350 case 0:
351 pbvUartObjectAscii.PBV_Header[0] = PBV_START_OF_FRAME;
352 pbvUartObjectAscii.PBV_Header[1] = pbvUartObjectAscii.Protocol_ID >> 8;
353 pbvUartObjectAscii.PBV_Header[2] = pbvUartObjectAscii.Protocol_ID & 0xff;
354 pbvUartObjectAscii.PBV_Header[3] = pbvUartObjectAscii.Length_in_Bytes >> 8;
355 pbvUartObjectAscii.PBV_Header[4] = pbvUartObjectAscii.Length_in_Bytes & 0xff;
356 pbvUartObjectAscii.Offset = 0;
357 pbvUartObjectAscii.UART_Frame_State = 1;
358 uartActiveTxAscii = true;
359 retval = PBV_MESSAGE_TRANSMITTING;
360 break;
361
362 case 1: //transfer header
363 if (PBV_ReadyToSend())// && (MsCounter_++ > 1)) //@ftx
364 {
365 temp = pbvUartObjectAscii.PBV_Header[pbvUartObjectAscii.Offset];
366 PBV_Write(temp);
367 pbvUartObjectAscii.Offset++;
368 if (pbvUartObjectAscii.Offset > 4)
369 {
370 pbvUartObjectAscii.Offset = 0;
371 pbvUartObjectAscii.UART_Frame_State = 2;
372 retval = PBV_MESSAGE_TRANSMITTING; // Why is this here?
373 }
374 }
375 else
376 {
377 retval = PBV_MESSAGE_TRANSMITTING;
378 }
379 break;
380
381 case 2: //transfer data
382 if (PBV_ReadyToSend())// && (MsCounter_++ > 1)) //@ftx
383 {
384 PBV_Write(pbvUartObjectAscii.data[pbvUartObjectAscii.Offset]);
385 pbvUartObjectAscii.Offset++;
386 if (pbvUartObjectAscii.Offset >= pbvUartObjectAscii.Length_in_Bytes)
387 {
388 pbvUartObjectAscii.UART_Frame_State = 3;
389 pbvUartObjectAscii.Offset = 0;
390 retval = PBV_MESSAGE_TRANSMITTING;
391 }
392 }
393 else
394 {
395 retval = PBV_MESSAGE_TRANSMITTING;
396 }
397 break;
398
399 case 3: //calculate CRC make end header
400 //IMPORTANT calculate crc before reusing PBV_HEADER
401 pbvUartObjectAscii.PBV_Header[0] = 0; //
402 pbvUartObjectAscii.PBV_Header[1] = 0; //add crc
403 pbvUartObjectAscii.PBV_Header[2] = PBV_END_OF_FRAME;
404 pbvUartObjectAscii.UART_Frame_State = 4;
405 retval = PBV_MESSAGE_TRANSMITTING;
406 break;
407
408 case 4: //transfer header
409 if (PBV_ReadyToSend())
410 {
411 PBV_Write(pbvUartObjectAscii.PBV_Header[pbvUartObjectAscii.Offset]);
412 pbvUartObjectAscii.Offset++;
413 if (pbvUartObjectAscii.Offset >= 3)
414 {
415 pbvUartObjectAscii.UART_Frame_State = 0;
416 pbvUartObjectAscii.Offset = 0;
417 uartActiveTxAscii = false;
418 retval = PBV_MESSAGE_TRANSMITTED;
419 }
420 }
421 else
422 {
423 retval = PBV_MESSAGE_TRANSMITTING;
424 }
425 break;
426
427 default:
428 retval = PBV_MESSAGE_TRANSMITTING;
429 break;
430 }
431 }
432 return retval;
433}

600uint16_t PBV_Calculate_CRC(uint8_t *message, uint16_t length)
601{
602 //CRC_Initialize();
603 uint8_t message_odd = (uint8_t) (length & 0x01);
604 uint16_t length_t = 0;
605 uint16_t i, j;
606 //uint16_t * message16bit;
607 uint32_t trycount = 0;
608 uint16_t resultCRC = 0;
609 uint16_t message16bit[PBV_RCV_DATABUFFER_SIZE];
610
611 //calCRCTEMP = 0; //
612
613 if (message_odd)
614 length_t = (length & 0xFFFE);
615 else
616 length_t = length;
617
618 for (i = 0, j = 0; i < length_t; i += 2, j++)
619 {
620 //*(message16bit + j) = (message[i + 1 ] << 8 ) | message[i] ;
621 message16bit[j] = (message[i + 1 ] << 8) | message[i];
622 }
623
624 CRC_CalculateBufferStart(message16bit, length_t);
625
626 while (!CRC_CalculationIsDone() && trycount < 2000)
627 {
628 CRC_Tasks();
629 trycount++;
630 }
631
632 resultCRC = CRC_CalculationResultReverseGet();
633
634 if (message_odd)
635 {
636 resultCRC ^= message[length - 1];
637 for (i = 0; i < 8; ++i)
638 {
639 if (resultCRC & 0x0001)
640 {
641 resultCRC = (resultCRC >> 1)^ PBV_CRC_POLYNOM_REV;
642 }
643 else
644 {
645 resultCRC = (resultCRC >> 1);
646 }
647 }
648 }
649 // clearing shift registers.
650 CRCWDATL = 0x0;
651 CRCWDATH = 0x0;
652
653 return resultCRC;
654}
