147void App_PBV_DAB_Task_10ms(void)
148{
149 // RX handler
150 if (appPbvDabRxPtr->PBV_Message_State == PBV_MESSAGE_RECEIVED)
151 {
152 App_Read_Received_From_PBV(appPbvDabRxPtr);
153 protocolID(appPbvDabRxPtr->PBV_Protcol_ID, appPbvDabRxPtr->Length, appPbvDabRxPtr->Data_Buffer);
154
155 // msg read. Read another
156 App_Receive_From_PBV(appPbvDabRxPtr);
157 }
159 if (++TickCounter > 11)
160 {
161 App_PBV_DAB_Build_Frame();
162 App_Send_To_PBV(appPbvDabTxPtr);
163 TickCounter = 0;
164 }

228void App_PBV_DAB_Build_Frame()
229{
230 // Power Board Visualizer can only mask single bits
231 // so create maskable word that can be parsed by GUI by sending the
232 // "2 ^ (state)" instead of "state"
233 BufferSixteenTx[0] = 1<<(Dev_PwrCtrl_Get_State());
234
235 // send back one "flag word" which combines fault and status and enable control flag
236 uint16_t enabled = Dev_PwrCtrl_Get_EnableFlag();
237 uint16_t fault_flags = Dev_Fault_GetFlags();
238 uint16_t status_flags = Dev_PwrCtrl_Get_Status();
239 uint16_t flag_word = enabled + ((status_flags & 0x0003)<<1) + (fault_flags<<3);
240
241 BufferSixteenTx[1] = flag_word;
242 BufferSixteenTx[2] = Dev_PwrCtrl_GetAveraging_Vprim(); //Dev_PwrCtrl_GetAdc_Vpri();
243 BufferSixteenTx[3] = Dev_PwrCtrl_GetAveraging_Vsec();//Dev_PwrCtrl_GetAdc_Vsec();
244 BufferSixteenTx[4] = Dev_PwrCtrl_GetAdc_Ipri_ct();
245 BufferSixteenTx[5] = Dev_PwrCtrl_GetAdc_Isec_ct();
246 BufferSixteenTx[6] = Dev_PwrCtrl_GetAveraging_Isec();//Dev_PwrCtrl_GetAdc_Isec_avg();
247 BufferSixteenTx[7] = Temperature + 40;//Dev_PwrCtrl_GetAdc_Temperature();
248 BufferSixteenTx[8] = Dev_PwrCtrl_GetAdc_Vrail_5V();
249 BufferSixteenTx[9] = Dev_PwrCtrl_GetPhase_P2SDegree();//devFanDataPtr->CurrentSpeedRaw;
250 BufferSixteenTx[10] = Dev_PwrCtrl_Get_DbgValue();//devFanDataPtr->CurrentSpeedPercent;
251 BufferSixteenTx[11] = Temperature;
252 BufferSixteenTx[12] = Dev_PwrCtrl_Get_Period();
253 BufferSixteenTx[13] = Dev_PwrCtrl_Get_PwmprdTarget();
254 BufferSixteenTx[14] = Dev_PwrCtrl_Get_SecPower();
255
256 PBV_Change_from_Sixteen_to_Eight(BufferSixteenTx, BufferEightTx, 18);
257
258 appPbvDabTxPtr->Data_Buffer = BufferEightTx;
259 appPbvDabTxPtr->Length = 18 * 2 ;
260}

271void App_PBV_DAB_Process_Rx_Data(uint16_t * data)
272{
273 uint16_t cmd_id = data[0];
274 uint16_t control_word = data[1];
275 switch (cmd_id)
276 {
277 case PBV_CMD_ID_DAB_ON_OFF: {
278 // turn DAB on or off
279 if (control_word < 2) {// should be 0 or 1
280
281 bool enable = (bool)control_word;
282 Dev_PwrCtrl_SetEnable(enable);
283
284 if(enable==false)
285 {
286 Dev_PwrCtrl_SetState(control_word);
287 }
288 }
289 break;
290 }
291 case PBV_CMD_ID_FREQ_CHANGE: {
292 // change target frequency
293 if ((control_word <= MAX_PWM_PERIOD) && (control_word >= MIN_PWM_PERIOD)) {
294 Dev_PwrCtrl_SetPeriodTarget(control_word);
295 // when Frequency is changed, control phase will be zero
296 uint16_t controlPhase = 0;
297 Dev_PwrCtrl_SetPhaseTarget(controlPhase);
298 }
299 break;
300 }
301 case PBV_CMD_ID_ILOOP_REF_SET: {
302 if (control_word < 32767) //TODO: put in proper check here!
303 {
304 Dev_PwrCtrl_SetIReference(control_word);
305 }
306 break;
307 }
308 case PBV_CMD_ID_VLOOP_REF_SET: {
309 if (control_word < 32767) //TODO: put in proper check here!
310 {
311 Dev_PwrCtrl_SetVSecReference(control_word);
312 }
313 break;
314 }
315
316 case PBV_CMD_ID_PLOOP_REF_SET: {
317 if (control_word < 32767) //TODO: put in proper check here!
318 {
319 Dev_PwrCtrl_SetPwrReference(control_word);
320 }
321 break;
322 }
323
324 case PBV_CMD_ID_P2S_PHASE_TARGET: {
325 // change target phase
326 #if(OPEN_LOOP_PBV == false)
327 uint16_t controlPhase = (uint16_t)(control_word);
328 Dev_PwrCtrl_SetP2SPhaseTarget(controlPhase);
329 #endif
330 break;
331 }
332
333 case PBV_CMD_ID_PHASE_CHANGE: {
334 // change target phase
335 #if(OPEN_LOOP_PBV == true)
336 uint16_t controlPhase = (uint16_t)((control_word)* PHASE_180_SCALER * (Dev_PwrCtrl_Get_DutyCycle()));
337 Dev_PwrCtrl_SetPhaseTarget(controlPhase);
338 #endif
339 break;
340 }
341 case PBV_CMD_ID_FAN_SPEED: {
342 Dev_Fan_Set_Speed(control_word);
343 break;
344 }
345 case PBV_CMD_ID_PRI_OVP_TEST: {
346 Dev_Fault_SetPriOVPThreshold(control_word);
347 break;
348 }
349 case PBV_CMD_ID_SEC_OVP_TEST: {
350 Dev_Fault_SetSecOVPThreshold(control_word);
351 break;
352 }
353 case PBV_CMD_ID_IPRI_TEST: {
354 Dev_Fault_SetIPrimaryThreshold(control_word);
355 break;
356 }
357 case PBV_CMD_ID_ISEC_TEST:{
358 Dev_Fault_SetISecondaryThreshold(control_word);
359 break;
360 }
361
362 default:
363 break;
364 }
365}
