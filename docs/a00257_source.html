219uint8_t PBV_UART_Receive_from_GUI()
220{
221 static uint16_t rcv_data_index = 0;
222 static uint16_t rcv_CRC = 0;
223 static uint16_t rcv_timeout = 0;
224 static uint8_t rcv_copy_for_CRC[PBV_RCV_DATABUFFER_SIZE + PBV_HEADER_SIZE];
225 uint8_t data;
226
227 while (1)
228 {
229 if (PBV_IsRxReady() == false)
230 {
231 if (++rcv_timeout >= 10000)
232 {
233 rcv_timeout = 0;
234 pbvUartObjectRx.UART_Frame_State = RCV_WAIT_FOR_STARTBYTE;
235 }
236 return PBV_MESSAGE_READY_TO_RECEIVE;
237 }
238 // ok, we have some data coming in, lets read and process
239 rcv_timeout = 0;
240 data = PBV_Read();
241 switch (pbvUartObjectRx.UART_Frame_State)
242 {
243 case RCV_WAIT_FOR_STARTBYTE:
244 if (data == PBV_START_OF_FRAME)
245 {
246 pbvUartObjectRx.UART_Frame_State = RCV_READ_ID_HIGHBYTE;
247 pbvUartObjectRx.offset = 0;
248 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.offset] = data;
249 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
250 rcv_data_index++; // for CRC
251 pbvUartObjectRx.offset++;
252 }
253 return PBV_STATE_RECEIVING;
254
255 case RCV_READ_ID_HIGHBYTE:
256 pbvUartObjectRx.Protocol_ID = data << 8;
257 pbvUartObjectRx.UART_Frame_State = RCV_READ_ID_LOWBYTE;
258 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.offset] = data;
259 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
260 rcv_data_index++; // for CRC
261 pbvUartObjectRx.offset++;
262 return PBV_STATE_RECEIVING;
263
264 case RCV_READ_ID_LOWBYTE:
265 pbvUartObjectRx.Protocol_ID |= data;
266 pbvUartObjectRx.UART_Frame_State = RCV_READ_LENGTH_HIGHBYTE;
267 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.offset] = data;
268 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
269 rcv_data_index++; // for CRC
270 pbvUartObjectRx.offset++;
271 return PBV_STATE_RECEIVING;
272
273 case RCV_READ_LENGTH_HIGHBYTE:
274 // clear CRC flag, for now, as the firmware doesn't support it
275 // With PBV version 2.0.0, dsPIC cannot receive
276 // messages from the GUI if this flag is not cleared
277 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.offset] = data;
278 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
279 rcv_data_index++; // for CRC
280 data = data & 0x7F; // clear bit 7 (CRC flag)
281 pbvUartObjectRx.Length_in_Bytes = data << 8;
282 pbvUartObjectRx.UART_Frame_State = RCV_READ_LENGTH_LOWBYTE;
283 pbvUartObjectRx.offset++;
284 return PBV_STATE_RECEIVING;
285
286 case RCV_READ_LENGTH_LOWBYTE:
287 pbvUartObjectRx.PBV_Header[pbvUartObjectRx.offset] = data;
288 pbvUartObjectRx.Length_in_Bytes |= data;
289 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
290 rcv_data_index++; // for CRC
291 pbvUartObjectRx.UART_Frame_State = RCV_READ_DATA;
292 pbvUartObjectRx.offset = 0;
293 return PBV_STATE_RECEIVING;
294
295 case RCV_READ_DATA:
296 if (pbvUartObjectRx.offset < PBV_RCV_DATABUFFER_SIZE + PBV_HEADER_SIZE ) // MAX
297 {
298 pbvUartObjectRx.data[pbvUartObjectRx.offset] = data;
299 pbvUartObjectRx.offset++;
300
301 rcv_copy_for_CRC[rcv_data_index] = data; // for CRC
302 rcv_data_index++; // for CRC
303 }
304 if (pbvUartObjectRx.offset >= pbvUartObjectRx.Length_in_Bytes) //are we finished receiving data???
305 {
306 pbvUartObjectRx.UART_Frame_State = RCV_READ_CRC_HIGHBYTE;
307 }
308 return PBV_STATE_RECEIVING;
309
310 case RCV_READ_CRC_HIGHBYTE:
311 rcv_CRC = data << 8;
312 pbvUartObjectRx.UART_Frame_State = RCV_READ_CRC_LOWBYTE;
313 return PBV_STATE_RECEIVING;
314
315 case RCV_READ_CRC_LOWBYTE:
316 rcv_CRC |= data;
317 pbvUartObjectRx.UART_Frame_State = RCV_READ_EOF;
318 return PBV_STATE_RECEIVING;
319
320 case RCV_READ_EOF:
321 if (data == 0x0d)
322 {
324 PBV_calculated_CRC = PBV_Calculate_CRC(rcv_copy_for_CRC, pbvUartObjectRx.Length_in_Bytes + PBV_HEADER_SIZE);
325 if (PBV_calculated_CRC == rcv_CRC)
326 {
327 rcv_data_index = 0;
328 rcv_CRC = 0;
329 PBV_calculated_CRC = 0;
330 pbvUartObjectRx.UART_Frame_State = RCV_MESSAGE_RECEIVED;
331 return PBV_MESSAGE_RECEIVED;
332 }
333 else {
334 rcv_data_index = 0; //for CRC
335 return PBV_MESSAGE_RX_ERROR;
336 }
337 }
338 }
339 }
340}

351uint8_t PBV_UART_Transmit_Ascii_to_GUI()
352{
353 uint8_t retval = PBV_MESSAGE_TX_ERROR;
354 uint8_t temp;
355 if (uartActiveTx == true)
356 {
357 retval = PBV_MESSAGE_TX_ERROR;
358 }
359 else
360 {
361 switch (pbvUartObjectAscii.UART_Frame_State)
362 {
363 case 0:
364 pbvUartObjectAscii.PBV_Header[0] = PBV_START_OF_FRAME;
365 pbvUartObjectAscii.PBV_Header[1] = pbvUartObjectAscii.Protocol_ID >> 8;
366 pbvUartObjectAscii.PBV_Header[2] = pbvUartObjectAscii.Protocol_ID & 0xff;
367 pbvUartObjectAscii.PBV_Header[3] = pbvUartObjectAscii.Length_in_Bytes >> 8;
368 pbvUartObjectAscii.PBV_Header[4] = pbvUartObjectAscii.Length_in_Bytes & 0xff;
369 pbvUartObjectAscii.offset = 0;
370 pbvUartObjectAscii.UART_Frame_State = 1;
371 uartActiveTxAscii = true;
372 retval = PBV_MESSAGE_TRANSMITTING;
373 break;
374
375 case 1: //transfer header
376 if (PBV_ReadyToSend())// && (MsCounter_++ > 1)) //@ftx
377 {
378 temp = pbvUartObjectAscii.PBV_Header[pbvUartObjectAscii.offset];
379 PBV_Write(temp);
380 pbvUartObjectAscii.offset++;
381 if (pbvUartObjectAscii.offset > 4)
382 {
383 pbvUartObjectAscii.offset = 0;
384 pbvUartObjectAscii.UART_Frame_State = 2;
385 retval = PBV_MESSAGE_TRANSMITTING; // Why is this here?
386 }
387 }
388 else
389 {
390 retval = PBV_MESSAGE_TRANSMITTING;
391 }
392 break;
393
394 case 2: //transfer data
395 if (PBV_ReadyToSend())// && (MsCounter_++ > 1)) //@ftx
396 {
397 PBV_Write(pbvUartObjectAscii.data[pbvUartObjectAscii.offset]);
398 pbvUartObjectAscii.offset++;
399 if (pbvUartObjectAscii.offset >= pbvUartObjectAscii.Length_in_Bytes)
400 {
401 pbvUartObjectAscii.UART_Frame_State = 3;
402 pbvUartObjectAscii.offset = 0;
403 retval = PBV_MESSAGE_TRANSMITTING;
404 }
405 }
406 else
407 {
408 retval = PBV_MESSAGE_TRANSMITTING;
409 }
410 break;
411
412 case 3: //calculate CRC make end header
413 //IMPORTANT calculate crc before reusing PBV_HEADER
414 pbvUartObjectAscii.PBV_Header[0] = 0; //
415 pbvUartObjectAscii.PBV_Header[1] = 0; //add crc
416 pbvUartObjectAscii.PBV_Header[2] = PBV_END_OF_FRAME;
417 pbvUartObjectAscii.UART_Frame_State = 4;
418 retval = PBV_MESSAGE_TRANSMITTING;
419 break;
420
421 case 4: //transfer header
422 if (PBV_ReadyToSend())
423 {
424 PBV_Write(pbvUartObjectAscii.PBV_Header[pbvUartObjectAscii.offset]);
425 pbvUartObjectAscii.offset++;
426 if (pbvUartObjectAscii.offset >= 3)
427 {
428 pbvUartObjectAscii.UART_Frame_State = 0;
429 pbvUartObjectAscii.offset = 0;
430 uartActiveTxAscii = false;
431 retval = PBV_MESSAGE_TRANSMITTED;
432 }
433 }
434 else
435 {
436 retval = PBV_MESSAGE_TRANSMITTING;
437 }
438 break;
439
440 default:
441 retval = PBV_MESSAGE_TRANSMITTING;
442 break;
443 }
444 }
445 return retval;
446}

624uint16_t PBV_Calculate_CRC(uint8_t *message, uint16_t length)
625{
626 //CRC_Initialize();
627 uint8_t message_odd = (uint8_t) (length & 0x01);
628 uint16_t length_t = 0;
629 uint16_t i, j;
630 //uint16_t * message16bit;
631 uint32_t trycount = 0;
632 uint16_t resultCRC = 0;
633 uint16_t message16bit[PBV_RCV_DATABUFFER_SIZE];
634
635 //calCRCTEMP = 0; //
636
637 if (message_odd)
638 length_t = (length & 0xFFFE);
639 else
640 length_t = length;
641
642 for (i = 0, j = 0; i < length_t; i += 2, j++)
643 {
644 //*(message16bit + j) = (message[i + 1 ] << 8 ) | message[i] ;
645 message16bit[j] = (message[i + 1 ] << 8) | message[i];
646 }
647
648 CRC_CalculateBufferStart(message16bit, length_t);
649
650 while (!CRC_CalculationIsDone() && trycount < 2000)
651 {
652 CRC_Tasks();
653 trycount++;
654 }
655
656 resultCRC = CRC_CalculationResultReverseGet();
657
658 if (message_odd)
659 {
660 resultCRC ^= message[length - 1];
661 for (i = 0; i < 8; ++i)
662 {
663 if (resultCRC & 0x0001)
664 {
665 resultCRC = (resultCRC >> 1)^ PBV_CRC_POLYNOM_REV;
666 }
667 else
668 {
669 resultCRC = (resultCRC >> 1);
670 }
671 }
672 }
673 // clearing shift registers.
674 CRCWDATL = 0x0;
675 CRCWDATH = 0x0;
676
677 return resultCRC;
678}
