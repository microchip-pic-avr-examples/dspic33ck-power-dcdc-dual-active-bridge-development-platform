77static void __inline__ Dev_SensorOffsetCal(SENSOR_OFFSET_CAL_t* sensor, uint16_t adcReading)
78{
79 // current sensor offset is not measured
80 sensor->Accumulator += adcReading;
81 if (++sensor->MeasurementCounter >= SENSOR_OFFSET_NUM_MEASUREMENTS)
82 {
83 // average to determine the offset
84 uint16_t offset = (uint16_t) (__builtin_divud(sensor->Accumulator, sensor->MeasurementCounter));
85
86 sensor->Accumulator = 0;
87 sensor->MeasurementCounter = 0;
88
89 // check if result is within acceptable limits
90 if ((offset < sensor->LimitHigh) && (offset > sensor->LimitLow))
91 {
92 // measured offset is reasonable. Update stored value and exit
93 sensor->Offset = offset;
94 sensor->CalibrationComplete = true;
95 }
96 }
97}
