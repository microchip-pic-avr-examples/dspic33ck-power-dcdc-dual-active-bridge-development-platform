230void App_PBV_DAB_Build_Frame()
231{
232 // Power Board Visualizer can only mask single bits
233 // so create maskable word that can be parsed by GUI by sending the
234 // "2 ^ (state)" instead of "state"
235 buffer_sixteen_tx[0] = 1<<(Dev_PwrCtrl_Get_State());
236
237 // send back one "flag word" which combines fault and status and enable control flag
238 uint16_t enabled = Dev_PwrCtrl_Get_EnableFlag();
239 uint16_t fault_flags = Dev_Fault_GetFlags();
240 uint16_t status_flags = Dev_PwrCtrl_Get_Status();
241 uint16_t current_sensor_cal_flag = Dev_CurrentSensor_Get_CalibrationStatus();
242 uint16_t flag_word = enabled + ((status_flags & 0x0003)<<1) + (fault_flags<<3);
243
244 buffer_sixteen_tx[1] = flag_word;
245 buffer_sixteen_tx[2] = Dev_PwrCtrl_GetAdc_Vpri();
246 buffer_sixteen_tx[3] = Dev_PwrCtrl_GetAveraging_Vsec();//Dev_PwrCtrl_GetAdc_Vsec();
247 buffer_sixteen_tx[4] = Dev_PwrCtrl_GetAdc_Ipri_ct();
248 buffer_sixteen_tx[5] = Dev_PwrCtrl_GetAdc_Isec_ct();
249 buffer_sixteen_tx[6] = Dev_PwrCtrl_GetAveraging_Isec();//Dev_PwrCtrl_GetAdc_Isec_avg();
250 buffer_sixteen_tx[7] = temperature + 40;//Dev_PwrCtrl_GetAdc_Temperature();
251 buffer_sixteen_tx[8] = Dev_PwrCtrl_GetAdc_Vrail_5V();
252 buffer_sixteen_tx[9] = devFanDataPtr->CurrentSpeedRaw;
253 buffer_sixteen_tx[10] = devFanDataPtr->CurrentSpeedPercent;
254 buffer_sixteen_tx[11] = temperature;
255 buffer_sixteen_tx[12] = Dev_PwrCtrl_Get_Period();
256 buffer_sixteen_tx[13] = Dev_PwrCtrl_Get_PwmprdTarget();
257 buffer_sixteen_tx[14] = Dev_PwrCtrl_Get_SecPower();
258
259 PBV_Change_from_Sixteen_to_Eight(buffer_sixteen_tx, buffer_eight_tx, 18);
260
261 App_PBV_DAB_TX_Ptr->Data_Buffer = buffer_eight_tx;
262 App_PBV_DAB_TX_Ptr->Length = 18 * 2 ;
263}

274void App_PBV_DAB_Process_Rx_Data(uint16_t * data)
275{
276 uint16_t cmd_id = data[0];
277 uint16_t control_word = data[1];
278 switch (cmd_id)
279 {
280 case PBV_CMD_ID_DAB_ON_OFF: {
281 // turn DAB on or off
282 if (control_word < 2) {// should be 0 or 1
283
284 bool enable = (bool)control_word;
285 Dev_PwrCtrl_SetEnable(enable);
286 if(enable==false)
287 {
288 Dev_PwrCtrl_SetIReference(0);
289 Dev_PwrCtrl_SetVSecReference(0);
290 Dev_PwrCtrl_SetPwrReference(0);
291 }
292 }
293 break;
294 }
295 case PBV_CMD_ID_DAB_RESET: {
296 Dev_PwrCtrl_SetState(control_word);
297 break;
298 }
299 case PBV_CMD_ID_FREQ_CHANGE: {
300 // change target frequency
301 if ((control_word <= MAX_PWM_PERIOD) && (control_word >= MIN_PWM_PERIOD)) {
302 Dev_PwrCtrl_SetPeriodTarget(control_word);
303 // when Frequency is changed, control phase will be zero
304 uint16_t controlPhase = 0;
305 Dev_PwrCtrl_SetPhaseTarget(controlPhase);
306 }
307 break;
308 }
309 case PBV_CMD_ID_ILOOP_REF_SET: {
310 if (control_word < 32767) //TODO: put in proper check here!
311 {
312 Dev_PwrCtrl_SetIReference(control_word);
313 }
314 break;
315 }
316 case PBV_CMD_ID_VLOOP_REF_SET: {
317 if (control_word < 32767) //TODO: put in proper check here!
318 {
319 Dev_PwrCtrl_SetVSecReference(control_word);
320 }
321 break;
322 }
323
324 case PBV_CMD_ID_PLOOP_REF_SET: {
325 if (control_word < 32767) //TODO: put in proper check here!
326 {
327 Dev_PwrCtrl_SetPwrReference(control_word);
328 }
329 break;
330 }
331
332 case PBV_CMD_ID_PHASE_CHANGE: {
333 // change target phase
334 Nop();
335 Nop();
336 Nop();
337 uint16_t controlPhase = (uint16_t)((control_word)* PHASE_180_SCALER * (Dev_PwrCtrl_Get_DutyCycle()));
338 Dev_PwrCtrl_SetPhaseTarget(controlPhase);
339 Nop();
340 Nop();
341 Nop();
342
343 break;
344 }
345 case PBV_CMD_ID_FAN_SPEED: {
346 Dev_Fan_Set_Speed(control_word);
347 break;
348 }
349 case PBV_CMD_ID_PRI_OVP_TEST: {
350 Dev_Fault_SetPriOVPThreshold(control_word);
351 break;
352 }
353 case PBV_CMD_ID_SEC_OVP_TEST: {
354 Dev_Fault_SetSecOVPThreshold(control_word);
355 break;
356 }
357 case PBV_CMD_ID_IPRI_TEST: {
358 Dev_Fault_SetIPrimaryThreshold(control_word);
359 break;
360 }
361 case PBV_CMD_ID_ISEC_TEST:{
362 Dev_Fault_SetISecondaryThreshold(control_word);
363 break;
364 }
365 case PBV_CMD_ID_PWM_DTH:{
366 if (control_word < 600) //TODO: put in proper check here!
367 control_word = 600;
368 Nop();
369 Nop();
370 Nop();
371 Dev_PwrCtrl_SetDeadTimeHigh(control_word);
372 break;
373 }
374 case PBV_CMD_ID_PWM_DTL:{
375 if (control_word < 600) //TODO: put in proper check here!
376 control_word = 600;
377
378 Dev_PwrCtrl_SetDeadTimeLow(control_word);
379 break;
380 }
381
382 default:
383 break;
384 }
385}
