92void Dev_LED_On(uint8_t led_id)
93{
94 if (led_id >= NUM_OF_LEDS)
95 return;
96 if (ledstates[led_id] == LED_STATE_ON)
97 return;
98 ledstates[led_id] = LED_STATE_ON;
99 switch(led_id)
100 {
101 case LED_PIM_RED: LED_DP_PIM_on(); break;
102 case LED_BOARD_RED: LED_Board_Red_On(); break;
103 case LED_BOARD_GREEN: LED_Board_Green_On(); break;
104 default: Nop(); break;
105 }
106}

222void Dev_LED_Task_100ms()
223{
224 uint8_t led_index=0;
225 uint8_t timermax=0;
226
227 for (led_index = 0; led_index < NUM_OF_LEDS; led_index++)
228 {
229 if (ledstates[led_index] > LED_STATE_ON)
230 {
231 if (ledstates[led_index] == LED_STATE_BLINK_SLOW)
232 timermax = BLINK_SLOW_INTERVAL;
233 else if (ledstates[led_index] == LED_STATE_BLINK_FAST)
234 timermax = BLINK_FAST_INTERVAL;
235 else // normal speed
236 timermax = BLINK_INTERVAL;
237 if (++ledtimer[led_index] >= timermax)
238 {
239 ledtimer[led_index] = 0;
240 switch(led_index)
241 {
242 case LED_PIM_RED: LED_DP_PIM_Toggle_State(); break;
243 case LED_BOARD_RED: LED_Board_Red_Toggle_State(); break;
244 case LED_BOARD_GREEN: LED_Board_Green_Toggle_State(); break;
245 default: Nop(); break;
246 }
247 }
248 }
249 }
