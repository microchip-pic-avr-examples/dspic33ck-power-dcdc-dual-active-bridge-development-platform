120bool FAULT_CheckMax(FAULT_OBJ_T *faultInput, int16_t faultSource, FAULT_CALLBACK callback)
121{
122
123 if ((faultSource >= faultInput->Val1_Threshold) && (faultInput->FaultActive == false))
124 {
125 faultInput->FaultCounter++;
126
127 if (faultInput->FaultCounter > faultInput->CounterSet)
128 {
129 faultInput->FaultLatch = true;
130
131 if (callback != NULL)
132 {
133 callback(); // Call Back function used to quickly shutdown system if needed when fault detected
134 }
135
136 faultInput->FaultActive = true;
137 faultInput->FaultCounter = 0U;
138 }
139 }
140 else if ((faultSource < (faultInput->Val1_Hysteresis)) && (faultInput->FaultActive == true))
141 {
142 faultInput->FaultCounter++;
143 if (faultInput->FaultCounter > faultInput->CounterReset)
144 {
145 faultInput->FaultCounter = 0U;
146 faultInput->FaultActive = false;
147 }
148 }
149 else
150 {
151 faultInput->FaultCounter = 0U;
152 }
153
154 return (faultInput->FaultActive);
155
156}

281bool FAULT_CheckRange(FAULT_OBJ_T *faultInput, int16_t inputValue, FAULT_CALLBACK callback)
282{
283 if (faultInput->FaultActive == false)
284 {
285 if ((inputValue <= faultInput->Val1_Threshold) ||
286 (inputValue >= faultInput->Val2_Threshold))
287 {
288 if (++faultInput->FaultCounter >= faultInput->CounterSet)
289 {
290 faultInput->FaultLatch = true;
291
292 if (callback != NULL)
293 {
294 callback(); // Call Back function used to quickly shutdown system if needed when fault detected
295 }
296 faultInput->FaultActive = true;
297 faultInput->FaultCounter = 0U;
298 }
299 }
300 else
301 {
302 faultInput->FaultCounter = 0U;
303 }
304 }
305 else
306 {
307 if ((inputValue > faultInput->Val1_Hysteresis) &&
308 (inputValue < faultInput->Val2_Hysteresis))
309 {
310 if (++faultInput->FaultCounter > faultInput->CounterReset)
311 {
312 faultInput->FaultCounter = 0U;
313 faultInput->FaultActive = false;
314 }
315 }
316 else
317 {
318 faultInput->FaultCounter = 0U;
319 }
320 }
321 return (faultInput->FaultActive);
