54void Dev_Fault_Handler(void)
55{
56 // Drive the fault pin to Low when Fault trips
57 FAULT_SetLow();
58
59 // Turn off PWM output
60 Dev_PwrCtrl_PWM_Disable(&dab);
61
62 // set the fault active bit
63 dab.Status.bits.FaultActive = 1;
64
65 // clear the running bit
66 dab.Status.bits.Running = 0;
67
68}

79void Dev_Fault_Initialize(void)
80{
81 // Initialize Primary Over Current Protection
82 FAULT_Init(&dab.Fault.Object.IPrimaryOCP, IPRI_OC_THRES_TRIG,
83 IPRI_OC_THRES_CLEAR, IPRI_OC_T_BLANK_TRIG, IPRI_OC_T_BLANK_CLEAR);
84
85 // Initialize Secondary Over Current Protection
86 FAULT_Init(&dab.Fault.Object.ISecondaryOCP, ISEC_OC_THRES_TRIG,
87 ISEC_OC_THRES_CLEAR, ISEC_OC_T_BLANK_TRIG, ISEC_OC_T_BLANK_CLEAR);
88
89 // Initialize Primary Over Voltage Protection
90 FAULT_Init(&dab.Fault.Object.VPrimaryOVP, VPRI_OV_THRES_TRIG,
91 VPRI_OV_THRES_CLEAR, VPRI_OV_T_BLANK_TRIG, VPRI_OV_T_BLANK_CLEAR);
92
93 // Initialize Secondary Over Voltage Protection
94 FAULT_Init(&dab.Fault.Object.VSecondaryOVP, VSEC_OV_THRES_TRIG,
95 VSEC_OV_THRES_CLEAR, VSEC_OV_T_BLANK_TRIG, VSEC_OV_T_BLANK_CLEAR);
96
97 // Initialize 5V Rail instability Protection
98 FAULT_Init(&dab.Fault.Object.VRail_5V, VRAIL_5V_UV_THRES_TRIG,
99 VRAIL_5V_UV_THRES_CLEAR, VRAIL_5V_UV_T_BLANK_TRIG, VRAIL_5V_UV_T_BLANK_CLEAR);
100
101 // Initialize Over Temperature Protection
102 FAULT_Init(&dab.Fault.Object.PowerSupplyOTP, MAX_TEMPERATURE_THRESHOLD_RAW,
103 OVER_TEMP_UPPER_THRESHOLD_WITH_HYST,FAULT_PERSISTENCE_COUNT_TEMP, FAULT_PERSISTENCE_COUNT_TEMP);
104
105 //ToDo: need to check this
106#if (FAULT_SHORT_CCT == true)
107 // Initialize short circuit fault protection with comparators
108 Dev_Fault_EnableShortCircuitProtection();
109#endif
110 // Clear the fault PCI for each PWM
111 Dev_Fault_ClearHardwareFaults();
112
113}

124void Dev_Fault_Execute(void)
125{
126 // secondary over current fault handler
127 #if (FAULT_ISEC_OC)
128 FAULT_CheckMax(&dab.Fault.Object.ISecondaryOCP, dab.Data.ISenseSecondary, &Dev_Fault_Handler);
129 #endif
130
131 // secondary over voltage fault handler
132 #if (FAULT_VSEC_OV)
133 FAULT_CheckMax(&dab.Fault.Object.VSecondaryOVP, dab.Data.VSecVoltage, &Dev_Fault_Handler);
134 #endif
135
136 // primary over current fault handler
137 #if(FAULT_IPRI_OC)
138 FAULT_CheckMax(&dab.Fault.Object.IPrimaryOCP, dab.Data.ISensePrimary, &Dev_Fault_Handler);
139 #endif
140
141 // primary over voltage fault handler
142 #if (FAULT_VPRI_OV)
143 FAULT_CheckMax(&dab.Fault.Object.VPrimaryOVP, dab.Data.VPriVoltage, &Dev_Fault_Handler);
144 #endif
145
146 // primary over voltage fault handler
147 #if (FAULT_VRAIL_5V)
148 FAULT_CheckMax(&dab.Fault.Object.VRail_5V, dab.Data.VRail_5V, &Dev_Fault_Handler);
149 #endif
150
151 // Hardware short circuit
152 if(CMP1_StatusGet() || CMP3_StatusGet()){
153 dab.Fault.Object.ISenseSCP.FaultActive = 1;
154 dab.Fault.Object.ISenseSCP.FaultLatch = 1;
155 }
156
157 // Identify the fault that trips
158 dab.Fault.FaultDetected = Dev_Fault_GetFlags();
159}

170void Dev_Fault_Reset(void)
171{
172 // Drive the fault pin to high to allow PWM signal drive
173 FAULT_SetHigh();
174
175 // Clear fault Objects FaultActive bit
176 dab.Fault.Object.ISenseSCP.FaultActive = 0;
177 dab.Fault.Object.IPrimaryOCP.FaultActive = 0;
178 dab.Fault.Object.ISecondaryOCP.FaultActive = 0;
179 dab.Fault.Object.VPrimaryOVP.FaultActive = 0;
180 dab.Fault.Object.VSecondaryOVP.FaultActive = 0;
181 dab.Fault.Object.ISenseSCP.FaultActive = 0;
182
183 // Clear fault Objects FaultLatch bit
184 dab.Fault.Object.ISenseSCP.FaultLatch = 0;
185 dab.Fault.Object.IPrimaryOCP.FaultLatch = 0;
186 dab.Fault.Object.ISecondaryOCP.FaultLatch = 0;
187 dab.Fault.Object.VPrimaryOVP.FaultLatch = 0;
188 dab.Fault.Object.VSecondaryOVP.FaultLatch = 0;
189 dab.Fault.Object.ISenseSCP.FaultLatch = 0;
190
191 //ToDo: remove this after testing LEB
192 // initialize thresholds of comparators used for short circuit protection
193 CMP_ISEC_SC_DACDataWrite(ISEC_SC_THRES_TRIG);
194 CMP_IPRI_SC_DACDataWrite(IPRI_SC_THRES_TRIG);
195
196 // Clear the fault PCI for each PWM
197 Dev_Fault_ClearHardwareFaults();
198}

209static void Dev_Fault_EnableShortCircuitProtection(void)
210{
211 // on dsPIC33CK DP-PIM:
212 // CMP1B used for short circuit protection on the secondary side
213 // CMP3B used for short circuit protection on the primary side
214
215 // initialize thresholds of comparators used for short circuit protection
216 CMP_ISEC_SC_DACDataWrite(ISEC_SC_THRES_TRIG);
217 CMP_IPRI_SC_DACDataWrite(IPRI_SC_THRES_TRIG);
218
219 // turn on comparator DACs
220 CMP_IPRI_SC_DACEnable();
221 CMP_ISEC_SC_DACEnable();
222
223}

233void Dev_Fault_ClearHardwareFaults(void)
234{
235 uint16_t pwmIndex;
236 for (pwmIndex = 1; pwmIndex <= 4; pwmIndex++)
237 {
238 PWM_FaultModeLatchClear(pwmIndex);
239 PWM_GeneratorEventStatusClear(pwmIndex, PWM_GENERATOR_INTERRUPT_FAULT);
240 }
241}
