143int8_t Dev_Temp_Get_Temperature_Celcius(void){
144
145 uint8_t index = 0;
146
147 TEMP_NTC_LUT_t point0;
148
149 devTempData.AdcAverage = Average_Temp_ADC_Samples();
150
151 if (devTempData.AdcAverage == 0) return 0;
152
153 if (devTempData.AdcAverage > 4056)
154 return NTC_lookuptable[0].temperature - 40;
155
156 if (devTempData.AdcAverage < 379)
157 return NTC_lookuptable[33].temperature - 40;
158
159 for (index = 0; index < 32; index++) {
160 if (devTempData.AdcAverage > NTC_lookuptable[index].ADC_Val) {
161 point0 = NTC_lookuptable[index];
162 break;
163 }
164 }
165 point0.temperatureCelsius=point0.temperature-40;
166 return point0.temperatureCelsius;
167}

196void Dev_Temp_Get_ADC_Sample(void)
197{

200 {
201 devTempData.BufferIndex = 0;
202 devTempData.BufferFull = 1;
203 }
204
205 // use API from power controller to read temperature
206 devTempData.AdcReading = Dev_PwrCtrl_GetAdc_Temperature();
207 devTempData.TempBuffer[devTempData.BufferIndex++] = devTempData.AdcReading;

220uint16_t Average_Temp_ADC_Samples(void) {

236uint16_t Temp_Calculate_Average(uint16_t * buffer, uint16_t length) {
237 uint16_t index = 0;
238 uint32_t sum = 0;
239 for (index = 0; index <length; index++)
240 sum +=buffer[index];
241 return (sum >> MAX_NUM_SAMPLES_TEMP_BUFFER_AS_EXP_OF_TWO);
242}
