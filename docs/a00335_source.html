100void Dev_Fan_Task_100ms(void)
101{
102 if (devFanData.OverrideFlag == 1)
103 {
104 ChangeSpeed = CHANGE_SPEED;
105 Override_Speed();
106 Calculate_Speed();
107 Update_Speed();
108 }
109 else if (++CurrentTickCount >= devFanData.Tick )
110 {
111 Calculate_Speed();
112 Update_Speed();
113 CurrentTickCount = 0;
114 }

162void Calculate_Speed(void)
163{
164 devFanData.TargetSpeedPercent -= (devFanData.TargetSpeedPercent % devFanData.StepSizePercent);
165
166 devFanData.TargetSpeedRaw = Convert_From_Percentage(devFanData.TargetSpeedPercent);
167
168 if (devFanData.CurrentSpeedRaw > (devFanData.TargetSpeedRaw )) // + dev_fan_data.hystersis_raw) )
169 {
170 devFanData.CurrentSpeedRaw -= Convert_From_Percentage(devFanData.StepSizePercent);
171 ChangeSpeed = CHANGE_SPEED;
172 }
173
174 else if (devFanData.CurrentSpeedRaw < (devFanData.TargetSpeedRaw )) // + dev_fan_data.hystersis_raw) )
175 {
176 devFanData.CurrentSpeedRaw += Convert_From_Percentage(devFanData.StepSizePercent) ;
177 ChangeSpeed = CHANGE_SPEED;
178 }
179 else
180 {
181 ChangeSpeed = NO_CHANGE;
182 }
183}

185
194void Update_Speed(void)
195{
196 if (ChangeSpeed != NO_CHANGE)
197 {
198 if (devFanData.CurrentSpeedRaw > devFanData.MaxSpeedRaw)
199 {
200 devFanData.CurrentSpeedRaw = devFanData.MaxSpeedRaw;
201 }
202 FAN_PWM_DutyCycleSet(devFanData.CurrentSpeedRaw);
203 devFanData.CurrentSpeedPercent = Convert_From_Raw(devFanData.CurrentSpeedRaw);
204 }
205}
