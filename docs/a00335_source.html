186void Dev_PwrCtrl_VComp_Initialize(void)
187{
188 //charger direction
189 VMC_2p2z.aCoefficients = &VMC_2p2zACoefficients[0]; // Set up pointer to derived coefficients
190 VMC_2p2z.bCoefficients = &VMC_2p2zBCoefficients[0]; // Set up pointer to derived coefficients
191 VMC_2p2z.controlHistory = &VMC_2p2zControlHistory[0]; // Set up pointer to controller history
192 VMC_2p2z.errorHistory = &VMC_2p2zErrorHistory[0]; // Set up pointer to error history
193
194 VMC_2p2z.preShift = DAB_SPVO_COMP_2P2Z_PRESHIFT;
195 VMC_2p2z.postScaler = DAB_SPVO_COMP_2P2Z_POSTSCALER;
196 VMC_2p2z.postShift = DAB_SPVO_COMP_2P2Z_POSTSHIFT;
197 VMC_2p2z.minOutput = DAB_SPVO_COMP_2P2Z_MIN_CLAMP;
198 VMC_2p2z.maxOutput = DAB_SPVO_COMP_2P2Z_MAX_CLAMP;
199
200 VMC_2p2zACoefficients[0] = DAB_SPVO_COMP_2P2Z_COEFF_A1;
201 VMC_2p2zACoefficients[1] = DAB_SPVO_COMP_2P2Z_COEFF_A2;
202 VMC_2p2zBCoefficients[0] = DAB_SPVO_COMP_2P2Z_COEFF_B0;
203 VMC_2p2zBCoefficients[1] = DAB_SPVO_COMP_2P2Z_COEFF_B1;
204 VMC_2p2zBCoefficients[2] = DAB_SPVO_COMP_2P2Z_COEFF_B2;
205
206 //explicit clear to 0
207 VMC_2p2zControlHistory[0] = 0;
208 VMC_2p2zControlHistory[1] = 0;
209 VMC_2p2zErrorHistory[0] = 0;
210 VMC_2p2zErrorHistory[1] = 0;
211 VMC_2p2zErrorHistory[2] = 0;
212
213 SMPS_Controller2P2ZInitialize(&VMC_2p2z);
214}
