200static void Calculate_Speed(void)
201{
202 devFanData.TargetSpeedPercent -= (devFanData.TargetSpeedPercent % devFanData.StepSizePercent);
203
204 devFanData.TargetSpeedRaw = Convert_From_Percentage(devFanData.TargetSpeedPercent);
205
206 if (devFanData.CurrentSpeedRaw > (devFanData.TargetSpeedRaw ))
207 {
208 devFanData.CurrentSpeedRaw -= Convert_From_Percentage(devFanData.StepSizePercent);
209 changeSpeed = CHANGE_SPEED;
210 }
211
212 else if (devFanData.CurrentSpeedRaw < (devFanData.TargetSpeedRaw ))
213 {
214 devFanData.CurrentSpeedRaw += Convert_From_Percentage(devFanData.StepSizePercent) ;
215 changeSpeed = CHANGE_SPEED;
216 }
217 else
218 {
219 changeSpeed = NO_CHANGE;
220 }
221}
