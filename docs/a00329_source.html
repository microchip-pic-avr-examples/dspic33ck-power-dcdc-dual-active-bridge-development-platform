66static void __inline__ Dev_SensorOffsetCal(SENSOR_OFFSET_CAL_t* sensor, uint16_t adcReading)
67{
68 // current sensor offset is not measured
69 sensor->Accumulator += adcReading;
70 if (++sensor->MeasurementCounter >= SENSOR_OFFSET_NUM_MEASUREMENTS)
71 {
72 // average to determine the offset
73 uint16_t offset = (uint16_t) (__builtin_divud(sensor->Accumulator, sensor->MeasurementCounter));
74
75 sensor->Accumulator = 0;
76 sensor->MeasurementCounter = 0;
77
78 // check if result is within acceptable limits
79 if ((offset < sensor->LimitHigh) && (offset > sensor->LimitLow))
80 {
81 // measured offset is reasonable. Update stored value and exit
82 sensor->Offset = offset;
83 sensor->CalibrationComplete = true;
84 }
85 }
86}

97void Dev_CurrentSensorOffsetCal(void)

101 Dev_SensorOffsetCal(&isecAvgCurrentSensor, ADC1_ConversionResultGet(ISEC_AVG));
102 }
