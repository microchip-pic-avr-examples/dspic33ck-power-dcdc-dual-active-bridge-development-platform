183static void Calculate_Speed(void)
184{
185 devFanData.TargetSpeedPercent -= (devFanData.TargetSpeedPercent % devFanData.StepSizePercent);
186
187 devFanData.TargetSpeedRaw = Convert_From_Percentage(devFanData.TargetSpeedPercent);
188
189 if (devFanData.CurrentSpeedRaw > (devFanData.TargetSpeedRaw )) // + dev_fan_data.hystersis_raw) )
190 {
191 devFanData.CurrentSpeedRaw -= Convert_From_Percentage(devFanData.StepSizePercent);
192 ChangeSpeed = CHANGE_SPEED;
193 }
194
195 else if (devFanData.CurrentSpeedRaw < (devFanData.TargetSpeedRaw )) // + dev_fan_data.hystersis_raw) )
196 {
197 devFanData.CurrentSpeedRaw += Convert_From_Percentage(devFanData.StepSizePercent) ;
198 ChangeSpeed = CHANGE_SPEED;
199 }
200 else
201 {
202 ChangeSpeed = NO_CHANGE;
203 }
204}

213static void Update_Speed(void)
214{
215 if (ChangeSpeed != NO_CHANGE)
216 {
217 if (devFanData.CurrentSpeedRaw > devFanData.MaxSpeedRaw)
218 {
219 devFanData.CurrentSpeedRaw = devFanData.MaxSpeedRaw;
220 }
221 FAN_PWM_DutyCycleSet(devFanData.CurrentSpeedRaw);
222 devFanData.CurrentSpeedPercent = Convert_From_Raw(devFanData.CurrentSpeedRaw);
223 }
224}
