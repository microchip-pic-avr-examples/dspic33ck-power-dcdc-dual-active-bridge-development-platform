46void PwrCtrl_PWM_Update(POWER_CONTROL_t* pcInstance)
47{
48 // The PWM Period bits [2:0] needs to be mask when using cascaded PWM setup
49 // (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
50 uint16_t PeriodMask = 0x7;
51
52 // Mask the calculated frequency bits [2:0] to make the cascaded/synchronous
53 // PWM scheme reliable (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
54 pcInstance->Pwm.ControlPeriod = pcInstance->Pwm.ControlPeriod & ~(PeriodMask);
55
56 // Calculate Duty Cycle for 50%
57 pcInstance->Pwm.ControlDutyCycle = (pcInstance->Pwm.ControlPeriod >> 1);
58
59 // Maximum Clamping for control phase
60 if(pcInstance->Pwm.ControlPhase > pcInstance->Pwm.ControlPeriod){
61 pcInstance->Pwm.ControlPhase = pcInstance->Pwm.ControlPeriod;
62 }
63
64 // Calculate primary to secondary phase as half of the control phase
65 uint16_t PrimarySecondaryPhase = (pcInstance->Pwm.ControlPhase >> 1);
66
67 // Compensate the added Dead-time
68 PrimarySecondaryPhase += pcInstance->Pwm.DeadTimeLow >> 1;
69
70 // Calculate the Bridge Delay ((Frequency / 2) - Primary to Secondary Phase + Control Phase)
71 // Note that in the cascaded PWM, the reference phase of the client PWM, is its trigger source
72 uint16_t PrimaryPhaseDelay = (pcInstance->Pwm.ControlDutyCycle - PrimarySecondaryPhase) +
73 pcInstance->Pwm.ControlPhase;
74
75 // Set the PWM trigger with the calculated PWM phases
76 PWM_TriggerCCompareValueSet(PWM_PRI_1, PrimarySecondaryPhase);
77 PWM_TriggerCCompareValueSet(PWM_SEC_1, PrimaryPhaseDelay);
78 PWM_TriggerCCompareValueSet(PWM_PRI_2, PrimarySecondaryPhase);
79
80 // Set the PWM Duty Cycle at 50% with the given Frequency
81 PWM_DutyCycleSet(PWM_PRI_1, pcInstance->Pwm.ControlDutyCycle);
82 PWM_DutyCycleSet(PWM_SEC_1, pcInstance->Pwm.ControlDutyCycle);
83 PWM_DutyCycleSet(PWM_PRI_2, pcInstance->Pwm.ControlDutyCycle);
84 PWM_DutyCycleSet(PWM_SEC_2, pcInstance->Pwm.ControlDutyCycle);
85
86 // Set the PWM Frequency
87 PWM_PeriodSet(PWM_PRI_1, pcInstance->Pwm.ControlPeriod);
88 PWM_PeriodSet(PWM_SEC_1, pcInstance->Pwm.ControlPeriod);
89 PWM_PeriodSet(PWM_PRI_2, pcInstance->Pwm.ControlPeriod);
90 PWM_PeriodSet(PWM_SEC_2, pcInstance->Pwm.ControlPeriod);
91
92 // Set the PWM Low DeadTime
93 PWM_DeadTimeLowSet(PWM_PRI_1, pcInstance->Pwm.DeadTimeLow);
94 PWM_DeadTimeLowSet(PWM_SEC_1, pcInstance->Pwm.DeadTimeLow);
95 PWM_DeadTimeLowSet(PWM_PRI_2, pcInstance->Pwm.DeadTimeLow);
96 PWM_DeadTimeLowSet(PWM_SEC_2, pcInstance->Pwm.DeadTimeLow);
97
98 // Set the PWM High DeadTime
99 PWM_DeadTimeHighSet(PWM_PRI_1, pcInstance->Pwm.DeadTimeHigh);
100 PWM_DeadTimeHighSet(PWM_SEC_1, pcInstance->Pwm.DeadTimeHigh);
101 PWM_DeadTimeHighSet(PWM_PRI_2, pcInstance->Pwm.DeadTimeHigh);
102 PWM_DeadTimeHighSet(PWM_SEC_2, pcInstance->Pwm.DeadTimeHigh);
103
104 // Set the Update bit of the last PWM in the cascaded approach to broadcast
105 // it to the other PWMs
106 PWM_SoftwareUpdateRequest(PWM_SEC_2);
107
108}

119void PwrCtrl_PWM_Enable(void)
120{
121 // Turn-On PWM outputs by disabling the output override
122 // on each high and low PWM output
123 PWM_OverrideHighDisable(PWM_PRI_1);
124 PWM_OverrideHighDisable(PWM_SEC_1);
125 PWM_OverrideHighDisable(PWM_PRI_2);
126 PWM_OverrideHighDisable(PWM_SEC_2);
127
128 PWM_OverrideLowDisable(PWM_PRI_1);
129 PWM_OverrideLowDisable(PWM_SEC_1);
130 PWM_OverrideLowDisable(PWM_PRI_2);
131 PWM_OverrideLowDisable(PWM_SEC_2);
132
133 // Set update request of the last PWM in the cascade
134 // to update all PWM registers
135 PWM_SoftwareUpdateRequest(PWM_SEC_2);
136
137}
