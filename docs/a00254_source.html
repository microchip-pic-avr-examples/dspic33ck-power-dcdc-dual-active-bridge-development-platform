136void App_PBV_DAB_Task_10ms(void)
137{
138 // RX handler
139 if (appPbvDabRxPtr->PBV_Message_State == PBV_MESSAGE_RECEIVED)
140 {
141 App_Read_Received_From_PBV(appPbvDabRxPtr);
142 protocolID(appPbvDabRxPtr->PBV_Protcol_ID, appPbvDabRxPtr->Length, appPbvDabRxPtr->Data_Buffer);
143
144 // msg read. Read another
145 App_Receive_From_PBV(appPbvDabRxPtr);
146 }
148 if (++tickCounter > 11)
149 {
150 App_PBV_DAB_Build_Frame();
151 App_Send_To_PBV(appPbvDabTxPtr);
152 tickCounter = 0;
153 }

214void App_PBV_DAB_Build_Frame()
215{
216 // Power Board Visualizer can only mask single bits
217 // so create maskable word that can be parsed by GUI by sending the
218 // "2 ^ (state)" instead of "state"
219 bufferSixteenTx[0] = 1<<(Dev_PwrCtrl_Get_State());
220
221 // send back one "flag word" which combines fault and status and enable control flag
222 uint16_t enabled = Dev_PwrCtrl_Get_EnableFlag();
223 uint16_t fault_flags = Fault_GetFlags();
224 uint16_t status_flags = Dev_PwrCtrl_Get_Status();
225 uint16_t flag_word = enabled + ((status_flags & 0x0003)<<1) + (fault_flags<<3);
226
227 bufferSixteenTx[1] = flag_word;
228 bufferSixteenTx[2] = Dev_PwrCtrl_GetAveraging_Vprim();
229 bufferSixteenTx[3] = Dev_PwrCtrl_GetAveraging_Vsec();
230 bufferSixteenTx[4] = PwrCtrl_GetAdc_Ipri_ct();
231 bufferSixteenTx[5] = PwrCtrl_GetAdc_Isec_ct();
232 bufferSixteenTx[6] = Dev_PwrCtrl_GetAveraging_Isec();
233 bufferSixteenTx[7] = temperature + 40;
234 bufferSixteenTx[8] = PwrCtrl_GetAdc_Vrail_5V();
235 bufferSixteenTx[9] = PwrCtrl_GetPhase_P2SDegree();
236 bufferSixteenTx[10] = devFanDataPtr->CurrentSpeedPercent;
237 bufferSixteenTx[11] = temperature;
238 bufferSixteenTx[12] = Dev_PwrCtrl_Get_Period();
239 bufferSixteenTx[13] = Dev_PwrCtrl_Get_PwmprdTarget();
240 bufferSixteenTx[14] = Dev_PwrCtrl_Get_SecPower();
241
242 PBV_Change_from_Sixteen_to_Eight(bufferSixteenTx, bufferEightTx, 18);
243
244 appPbvDabTxPtr->Data_Buffer = bufferEightTx;
245 appPbvDabTxPtr->Length = 18 * 2 ;
246}

257void App_PBV_DAB_Process_Rx_Data(uint16_t * data)
258{
259 uint16_t cmd_id = data[0];
260 uint16_t control_word = data[1];
261 switch (cmd_id)
262 {
263 case PBV_CMD_ID_DAB_ON: {
264 PwrCtrl_SetEnable(true);
265 break;
266 }
267 case PBV_CMD_ID_DAB_OFF: {
268 PwrCtrl_SetEnable(false);
269 break;
270 }
271 case PBV_CMD_ID_FREQ_CHANGE: {
272 // change target frequency
273 if ((control_word <= MAX_PWM_PERIOD) && (control_word >= MIN_PWM_PERIOD)) {
274 PwrCtrl_SetPeriodTarget(control_word);
275 // when Frequency is changed, control phase will be zero
276 uint16_t controlPhase = 0;
277 PwrCtrl_SetPhaseTarget(controlPhase);
278 }
279 break;
280 }
281 case PBV_CMD_ID_ILOOP_REF_SET: {
282 if (control_word < 32767) //TODO: put in proper check here!
283 {
284 PwrCtrl_SetIReference(control_word);
285 }
286 break;
287 }
288 case PBV_CMD_ID_VLOOP_REF_SET: {
289 if (control_word < 32767) //TODO: put in proper check here!
290 {
291 PwrCtrl_SetVSecReference(control_word);
292 }
293 break;
294 }
295
296 case PBV_CMD_ID_PLOOP_REF_SET: {
297 if (control_word < 32767) //TODO: put in proper check here!
298 {
299 PwrCtrl_SetPwrReference(control_word);
300 }
301 break;
302 }
303
304 case PBV_CMD_ID_P2S_PHASE_TARGET: {
305 // change target phase
306 #if(OPEN_LOOP_PBV == false)
307 uint16_t controlPhase = (uint16_t)(control_word);
308 PwrCtrl_SetP2SPhaseTarget(controlPhase);
309 #endif
310 break;
311 }
312
313 case PBV_CMD_ID_PHASE_CHANGE: {
314 // change target phase
315 #if(OPEN_LOOP_PBV == true)
316 uint16_t controlPhase = (uint16_t)((control_word)* PHASE_180_SCALER * (Dev_PwrCtrl_Get_DutyCycle()));
317 PwrCtrl_SetPhaseTarget(controlPhase);
318 #endif
319 break;
320 }
321 case PBV_CMD_ID_FAN_SPEED: {
322 Dev_Fan_Set_Speed(control_word);
323 break;
324 }
325
326 default:
327 break;
328 }
329}
