219void App_PBV_DAB_Build_Frame()
220{
221 // Power Board Visualizer can only mask single bits
222 // so create maskable word that can be parsed by GUI by sending the
223 // "2 ^ (state)" instead of "state"
224 bufferSixteenTx[0] = 1<<(Dev_PwrCtrl_Get_State());
225
226 // send back one "flag word" which combines fault and status and enable control flag
227 uint16_t enabled = Dev_PwrCtrl_Get_EnableFlag();
228 uint16_t fault_flags = Fault_GetFlags();
229 uint16_t status_flags = Dev_PwrCtrl_Get_Status();
230 uint16_t flag_word = enabled + ((status_flags & 0x0003)<<1) + (fault_flags<<3);
231
232 bufferSixteenTx[1] = flag_word;
233 bufferSixteenTx[2] = Dev_PwrCtrl_GetAveraging_Vprim();
234 bufferSixteenTx[3] = Dev_PwrCtrl_GetAveraging_Vsec();
235 bufferSixteenTx[4] = PwrCtrl_GetAdc_Ipri_ct();
236 bufferSixteenTx[5] = PwrCtrl_GetAdc_Isec_ct();
237 bufferSixteenTx[6] = Dev_PwrCtrl_GetAveraging_Isec();
238 bufferSixteenTx[7] = temperature + 40;
239 bufferSixteenTx[8] = PwrCtrl_GetAdc_Vrail_5V();
240 bufferSixteenTx[9] = PwrCtrl_GetPhase_P2SDegree();
241 bufferSixteenTx[10] = devFanDataPtr->CurrentSpeedPercent;
242 bufferSixteenTx[11] = temperature;
243 bufferSixteenTx[12] = Dev_PwrCtrl_Get_Period();
244 bufferSixteenTx[13] = Dev_PwrCtrl_Get_PwmprdTarget();
245 bufferSixteenTx[14] = Dev_PwrCtrl_Get_SecPower();
246
247 PBV_Change_from_Sixteen_to_Eight(bufferSixteenTx, bufferEightTx, 18);
248
249 appPbvDabTxPtr->Data_Buffer = bufferEightTx;
250 appPbvDabTxPtr->Length = 18 * 2 ;
251}

262void App_PBV_DAB_Process_Rx_Data(uint16_t * data)
263{
264 uint16_t cmd_id = data[0];
265 uint16_t control_word = data[1];
266 switch (cmd_id)
267 {
268 case PBV_CMD_ID_DAB_ON_OFF: {
269 // turn DAB on or off
270 if (control_word < 2) {// should be 0 or 1
271
272 bool enable = (bool)control_word;
273 PwrCtrl_SetEnable(enable);
274
275 if(enable==false)
276 {
277 Dev_PwrCtrl_SetState(control_word);
278 }
279 }
280 break;
281 }
282 case PBV_CMD_ID_FREQ_CHANGE: {
283 // change target frequency
284 if ((control_word <= MAX_PWM_PERIOD) && (control_word >= MIN_PWM_PERIOD)) {
285 PwrCtrl_SetPeriodTarget(control_word);
286 // when Frequency is changed, control phase will be zero
287 uint16_t controlPhase = 0;
288 PwrCtrl_SetPhaseTarget(controlPhase);
289 }
290 break;
291 }
292 case PBV_CMD_ID_ILOOP_REF_SET: {
293 if (control_word < 32767) //TODO: put in proper check here!
294 {
295 PwrCtrl_SetIReference(control_word);
296 }
297 break;
298 }
299 case PBV_CMD_ID_VLOOP_REF_SET: {
300 if (control_word < 32767) //TODO: put in proper check here!
301 {
302 PwrCtrl_SetVSecReference(control_word);
303 }
304 break;
305 }
306
307 case PBV_CMD_ID_PLOOP_REF_SET: {
308 if (control_word < 32767) //TODO: put in proper check here!
309 {
310 PwrCtrl_SetPwrReference(control_word);
311 }
312 break;
313 }
314
315 case PBV_CMD_ID_P2S_PHASE_TARGET: {
316 // change target phase
317 #if(OPEN_LOOP_PBV == false)
318 uint16_t controlPhase = (uint16_t)(control_word);
319 PwrCtrl_SetP2SPhaseTarget(controlPhase);
320 #endif
321 break;
322 }
323
324 case PBV_CMD_ID_PHASE_CHANGE: {
325 // change target phase
326 #if(OPEN_LOOP_PBV == true)
327 uint16_t controlPhase = (uint16_t)((control_word)* PHASE_180_SCALER * (Dev_PwrCtrl_Get_DutyCycle()));
328 PwrCtrl_SetPhaseTarget(controlPhase);
329 #endif
330 break;
331 }
332 case PBV_CMD_ID_FAN_SPEED: {
333 Dev_Fan_Set_Speed(control_word);
334 break;
335 }
336
337 default:
338 break;
339 }
340}
