145int8_t Dev_Temp_Get_Temperature_Celcius(void){
146
147 uint8_t index = 0;
148
149 TEMP_NTC_LUT_t point0;
150
151 devTempData.AdcAverage = Average_Temp_ADC_Samples();
152
153 if (devTempData.AdcAverage == 0) return 0;
154
155 if (devTempData.AdcAverage > 4056)
156 return ntcLookUpTable[0].temperature - 40;
157
158 if (devTempData.AdcAverage < 379)
159 return ntcLookUpTable[33].temperature - 40;
160
161 for (index = 0; index < 32; index++) {
162 if (devTempData.AdcAverage > ntcLookUpTable[index].ADC_Val) {
163 point0 = ntcLookUpTable[index];
164 break;
165 }
166 }
167 point0.temperatureCelsius=point0.temperature-40;
168 return point0.temperatureCelsius;
169}

191void Dev_Temp_Get_ADC_Sample(void)
192{

194 if (devTempData.BufferIndex >= (MAX_NUM_SAMPLES_TEMP_BUFFER - 1))
195 {
196 devTempData.BufferIndex = 0;
197 devTempData.BufferFull = 1;
198 }
199
200 // use API from power controller to read temperature
201 devTempData.AdcReading = PwrCtrl_GetAdc_Temperature();
202 devTempData.TempBuffer[devTempData.BufferIndex++] = devTempData.AdcReading;
203
204}

213static uint16_t Average_Temp_ADC_Samples(void) {
214 if (devTempData.BufferFull)
215 return Temp_Calculate_Average(devTempData.TempBuffer, MAX_NUM_SAMPLES_TEMP_BUFFER);
216 else
217 return 0;
218}

227static uint16_t Temp_Calculate_Average(uint16_t * buffer, uint16_t length) {
228 uint16_t index = 0;
229 uint32_t sum = 0;
230 for (index = 0; index <length; index++)
231 sum +=buffer[index];
232 return (sum >> MAX_NUM_SAMPLES_TEMP_BUFFER_AS_EXP_OF_TWO);
233}
