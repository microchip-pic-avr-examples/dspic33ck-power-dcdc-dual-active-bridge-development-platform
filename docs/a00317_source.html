40uint16_t Dev_Fault_GetFlags(void)
41{
42 uint16_t fault = 0;
43
44 // Concatenate the Fault status in one for PBV
45 fault = (dab.Fault.Object.IPrimaryOCP.FaultLatch << FLT_IPRI_OCP) +
46 (dab.Fault.Object.ISecondaryOCP.FaultLatch << FLT_ISEC_OCP) +
47 (dab.Fault.Object.ISenseSCP.FaultLatch << FLT_ISNS_SCP) +
48 (dab.Fault.Object.VPrimaryOVP.FaultLatch << FLT_VPRI_OVP) +
49 (dab.Fault.Object.VSecondaryOVP.FaultLatch << FLT_VSEC_OVP) +
50 (dab.Fault.Object.PowerSupplyOTP.FaultLatch << FLT_DAB_TEMP) +
51 (dab.Fault.Object.VRail_5V.FaultLatch << FLT_RAIL_5V);
52
53 return(fault);
54}
