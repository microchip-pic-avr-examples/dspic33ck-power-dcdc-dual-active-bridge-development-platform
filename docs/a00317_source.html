87int8_t Dev_Temp_Get_Temperature_Celcius(void){
88
89 TEMP_NTC_LUT_t point0;
90
91 if (devTempData.AdcAverage == 0) return 0;
92
93 if (devTempData.BufferFull)
94 point0.temperature = (__builtin_mulsu(TEMPERATURE_FACTOR, devTempData.AdcAverage) >> 15) + TEMPERATURE_OFFSET;
95
96 point0.temperatureCelsius = point0.temperature - TEMPERATURE_PBV_OFFSET_CELCIUS;
97
98 return point0.temperatureCelsius;

157static uint16_t Temp_Calculate_Average(uint16_t * buffer, uint16_t length) {
158 uint16_t index = 0;
159 uint32_t sum = 0;
160 for (index = 0; index <length; index++)
161 sum +=buffer[index];
162 return (sum >> MAX_NUM_SAMPLES_TEMP_BUFFER_AS_EXP_OF_TWO);
163}
