37void Dev_PwrCtrl_PWM_Update(POWER_CONTROL_t* pcInstance)
38{
39 // The PWM Period bits [2:0] needs to be mask when using cascaded PWM setup
40 // (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
41 uint16_t PeriodMask = 0x7;
42
43 // Mask the calculated frequency bits [2:0] to make the cascaded/synchronous
44 // PWM scheme reliable (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
45 pcInstance->Pwm.ControlPeriod = pcInstance->Pwm.ControlPeriod & ~(PeriodMask);
46
47 // Calculate Duty Cycle for 50%
48 pcInstance->Pwm.ControlDutyCycle = (pcInstance->Pwm.ControlPeriod >> 1);
49
50 // Maximum Clamping for control phase
51 if(pcInstance->Pwm.ControlPhase > pcInstance->Pwm.ControlPeriod){
52 pcInstance->Pwm.ControlPhase = pcInstance->Pwm.ControlPeriod;
53 }
54
55 // Calculate primary to secondary phase as half of the control phase
56 uint16_t PrimarySecondaryPhase = (pcInstance->Pwm.ControlPhase >> 1);
57
58 // Calculate the Bridge Delay ((Frequency / 2) - Primary to Secondary Phase + Control Phase)
59 // Note that in the cascaded PWM, the reference phase of the client PWM, is its trigger source
60 uint16_t PrimaryPhaseDelay = (pcInstance->Pwm.ControlDutyCycle - PrimarySecondaryPhase) +
61 pcInstance->Pwm.ControlPhase;
62
63 // Set the PWM trigger with the calculated PWM phases
64 PWM_TriggerCCompareValueSet(pcInstance->Pwm.Primary_1, PrimarySecondaryPhase);
65 PWM_TriggerCCompareValueSet(pcInstance->Pwm.Secondary_1, PrimaryPhaseDelay);
66 PWM_TriggerCCompareValueSet(pcInstance->Pwm.Primary_2, PrimarySecondaryPhase);
67
68 // Set the PWM Duty Cycle at 50% with the given Frequency
69 PWM_DutyCycleSet(pcInstance->Pwm.Primary_1, pcInstance->Pwm.ControlDutyCycle);
70 PWM_DutyCycleSet(pcInstance->Pwm.Secondary_1, pcInstance->Pwm.ControlDutyCycle);
71 PWM_DutyCycleSet(pcInstance->Pwm.Primary_2, pcInstance->Pwm.ControlDutyCycle);
72 PWM_DutyCycleSet(pcInstance->Pwm.Secondary_2, pcInstance->Pwm.ControlDutyCycle);
73
74 // Set the PWM Frequency
75 PWM_PeriodSet(pcInstance->Pwm.Primary_1, pcInstance->Pwm.ControlPeriod);
76 PWM_PeriodSet(pcInstance->Pwm.Secondary_1, pcInstance->Pwm.ControlPeriod);
77 PWM_PeriodSet(pcInstance->Pwm.Primary_2, pcInstance->Pwm.ControlPeriod);
78 PWM_PeriodSet(pcInstance->Pwm.Secondary_2, pcInstance->Pwm.ControlPeriod);
79
80 // Set the PWM High and Low DeadTime
81 // DeadTime parameter added for testing
82 PWM_DeadTimeLowSet(pcInstance->Pwm.Primary_1, pcInstance->Pwm.DeadTimeLow);
83 PWM_DeadTimeLowSet(pcInstance->Pwm.Secondary_1, pcInstance->Pwm.DeadTimeLow);
84 PWM_DeadTimeLowSet(pcInstance->Pwm.Primary_2, pcInstance->Pwm.DeadTimeLow);
85 PWM_DeadTimeLowSet(pcInstance->Pwm.Secondary_2, pcInstance->Pwm.DeadTimeLow);
86
87 // DeadTime parameter added for testing
88 PWM_DeadTimeHighSet(pcInstance->Pwm.Primary_1, pcInstance->Pwm.DeadTimeHigh);
89 PWM_DeadTimeHighSet(pcInstance->Pwm.Secondary_1, pcInstance->Pwm.DeadTimeHigh);
90 PWM_DeadTimeHighSet(pcInstance->Pwm.Primary_2, pcInstance->Pwm.DeadTimeHigh);
91 PWM_DeadTimeHighSet(pcInstance->Pwm.Secondary_2, pcInstance->Pwm.DeadTimeHigh);
92
93 // Set the Update bit of the last PWM in the cascaded approach to broadcast
94 // it to the other PWMs
95 PWM_SoftwareUpdateRequest(pcInstance->Pwm.Secondary_2);
96
97}

108void Dev_PwrCtrl_PWM_Enable(POWER_CONTROL_t* pcInstance)
109{
110
111 // Turn-On PWM outputs by disabling the output override
112 // on each high and low PWM output
113 PWM_OverrideHighDisable(pcInstance->Pwm.Primary_1);
114 PWM_OverrideHighDisable(pcInstance->Pwm.Primary_2);
115 PWM_OverrideHighDisable(pcInstance->Pwm.Secondary_1);
116 PWM_OverrideHighDisable(pcInstance->Pwm.Secondary_2);
117
118 PWM_OverrideLowDisable(pcInstance->Pwm.Primary_1);
119 PWM_OverrideLowDisable(pcInstance->Pwm.Primary_2);
120 PWM_OverrideLowDisable(pcInstance->Pwm.Secondary_1);
121 PWM_OverrideLowDisable(pcInstance->Pwm.Secondary_2);
122
123 // Set update request of the last PWM in the cascade
124 // to update all PWM registers
125 PWM_SoftwareUpdateRequest(pcInstance->Pwm.Secondary_2);
126
127}
