67void ControlLoop_Interrupt_CallBack(void)
68{
69 GPIO_1_SetHigh();
70
71 // Update the ADC data member
72 Dev_PwrCtrl_UpdateADConverterData();
73
74 // Execute the fault detection
75 Dev_Fault_Execute();
76
77 // Performs data averaging necessary for control loop
78 // also Enabling interleaved the Vloop (10KHz execution)
79 // and Ploop (10KHz execution) enable bit
80 Dev_PwrCtrl_10KHzVPLoopPrepareData();
81
82 #if(OPEN_LOOP_PBV == false)
83
84 if(dab.Status.bits.Running == 1){
85 // Execute Power Converter Control Loop
86 Dev_PwrCtrl_ControlLoopExecute(); }
87
88 // Measure Primary to Secondary phase in degrees
89 Dev_PwrCtrl_PrimToSecPHDegree();
90
91 // Adjust DeadTime based on Primary to Secondary phase in degrees
92 Dev_PwrCtrl_DeadTimeAdjust();
93
94 #if(PERIOD_MODULATION_DEMO == true)
95 Dev_PwrCtrl_PeriodModulator();
96 #endif
97
98 #endif
99
100 #if(DAC_DEBUG == true)
101 // This is for DAC debugging purposes
102// CMP2_DACDataWrite(variable_to_test_here);
103 #endif
104
105 #if (true == OPEN_LOOP_POTI)
106
107 // Connect TP11 -> Pot1 & TP12 -> Pot2 in digital Power Development Board
108 uint16_t Pot1 = dab.Data.ISecAverage;
109 uint16_t Pot2 = dab.Data.ISenseSecondary;
110
111 // Calculate the Frequency based on the Potentiometer 1 voltage
112 dab.Pwm.ControlPeriod = (uint16_t)(MIN_PWM_PERIOD + (Pot1 * ADC_PERIOD_RANGE));
113
114 // Change the control Phase with Potentiometer 2
115 dab.Pwm.ControlPhase = (uint16_t)(Pot2 * ADC_SCALER * (dab.Pwm.ControlPeriod >> 1));
116
117 #endif
118
119 // Update PWM Properties
120 Dev_PwrCtrl_PWM_Update(&dab);
121
122 // Enable the ADC sampling
123 ADC1_SoftwareTriggerEnable();
124
125 GPIO_1_SetLow();
126}
