82void Dev_PwrCtrl_UpdateADConverterData (void)
83{
84 // Enable the ADC sampling
85 ADC1_SoftwareTriggerEnable();
86
87 if(ADC1_IsConversionComplete(ISEC_CT))
88 dab.Data.ISenseSecondary = ADC1_ConversionResultGet(ISEC_CT);
89
90 if(ADC1_IsConversionComplete(IPRI_CT))
91 dab.Data.ISensePrimary = ADC1_ConversionResultGet(IPRI_CT);
92
93 if(ADC1_IsConversionComplete(VRAIL_5V))
94 dab.Data.VRail_5V = ADC1_ConversionResultGet(VRAIL_5V);
95
96 if(ADC1_IsConversionComplete(TEMP))
97 dab.Data.Temperature = ADC1_ConversionResultGet(TEMP);
98
99 if(ADC1_IsConversionComplete(VPRI)){
100 dab.Data.VPriVoltage = ADC1_ConversionResultGet(VPRI);
101 VprimAveraging.Accumulator += dab.Data.VPriVoltage;
102 VprimAveraging.Counter = VprimAveraging.Counter + 1;
103 }
104
105 if(ADC1_IsConversionComplete(VSEC)){
106 dab.Data.VSecVoltage = ADC1_ConversionResultGet(VSEC);
107 VsecAveraging.Accumulator += dab.Data.VSecVoltage;
108 VsecAveraging.Counter = VsecAveraging.Counter + 1;
109 }
110
111 if(ADC1_IsConversionComplete(ISEC_AVG)){
112 dab.Data.ISecAverage = ADC1_ConversionResultGet(ISEC_AVG);
113 dab.Data.ISecAverageRectified = abs(dab.Data.ISecAverage - dab.Data.ISecSensorOffset);
114 IsecAveraging.Accumulator += dab.Data.ISecAverageRectified;
115 IsecAveraging.Counter = IsecAveraging.Counter + 1;
116 }
117}

129void Dev_PwrCtrl_10KHzVPLoopPrepareData(void)
130{
131 static uint16_t cnt = 0;
132
133 //Interleave the execution of VLoop and PLoop control for 10KHz execution
134 if(++cnt == (VPLOOP_ILOOP_EXE_RATIO))
135 {
136 // Averaging of Secondary Voltage
137 VsecAveraging.AverageValue = (uint16_t)(__builtin_divud(VsecAveraging.Accumulator, VsecAveraging.Counter));
138 VsecAveraging.Accumulator = 0;
139 VsecAveraging.Counter = 0;
140
141 #if(OPEN_LOOP_PBV == false)
142 //Condition for control loop execution controlling the loop enable bit
143 if((dab.VLoop.Enable == false) && (VLoopInterleaveExec == true))
144 #endif
145 {
146 // Enable Vloop control
147 dab.VLoop.Enable = true;
148 dab.PLoop.Enable = false;
149
150 // Averaging of Primary Voltage
151 VprimAveraging.AverageValue = (uint16_t)(__builtin_divud(VprimAveraging.Accumulator, VprimAveraging.Counter));
152 VprimAveraging.Accumulator = 0;
153 VprimAveraging.Counter = 0;
154
155 // Compute the Adaptive Gain
156 Dev_PwrCtrl_AdaptiveGainUpdate();
157 }
158
159 #if(OPEN_LOOP_PBV == false)
160 else if((dab.PLoop.Enable == false) && (VLoopInterleaveExec == false))
161 #endif
162 {
163 // Enable PLoop control
164 dab.VLoop.Enable = false;
165 dab.PLoop.Enable = true;
166
167 // Averaging of Secondary Current
168 IsecAveraging.AverageValue = (uint16_t)(__builtin_divud(IsecAveraging.Accumulator, IsecAveraging.Counter));
169 IsecAveraging.Accumulator = 0;
170 IsecAveraging.Counter = 0;
171
172 // Bit-shift value used to perform input value normalization
173 // Scaled the feedback to Power (Watts in units)
174 uint32_t buf = (uint32_t)IsecAveraging.AverageValue *
175 (uint32_t)VsecAveraging.AverageValue * POWER_FACTOR;
176
177 // scale back the 14 bit from the POWER_RESOLUTION calculation
178 // to get the Watts value for Power Loop
179 buf >>= POWER_SCALER;
180
181 // Transfer to SecPower data member the power computation in [Watts]
182 dab.Data.SecPower = buf;
183
184 }
185 cnt = 0;
186 }
187}

200void Dev_PwrCtrl_ControlLoopExecute(void)
201{
202 // Execute the Voltage Loop Control
203 if((dab.VLoop.Enable == true) && (VLoopInterleaveExec == true))
204 {
205 VLoopInterleaveExec = false;
206
207 if(dab.PowerDirection == PWR_CTRL_CHARGING)
208 {
209 VMC_2p2z.KfactorCoeffsB = 0x7FFF;
210 VMC_2p2z.maxOutput = 0x7FFF;
211
212 // Bit-shift value used to perform input value normalization
213 dab.VLoop.Feedback = VsecAveraging.AverageValue << 3;
214 dab.VLoop.Reference = dab.VLoop.Reference << 3;
215
216 // Execute the Voltage Loop Control
217 SMPS_Controller2P2ZUpdate(&VMC_2p2z, &dab.VLoop.Feedback,
218 dab.VLoop.Reference, &dab.VLoop.Output);
219
220 // Reset the Vloop reference to its original scaling
221 dab.VLoop.Reference = dab.VLoop.Reference >> 3;
222 }
223 }
224
225 // Execute the Power Loop Control
226 if((dab.PLoop.Enable == true) && (VLoopInterleaveExec == false))
227 {
228 VLoopInterleaveExec = true;
229
230 dab.PLoop.Feedback = dab.Data.SecPower;
231 dab.PLoop.Reference = dab.PLoop.Reference;
232
233 // Execute the Power Loop Control
234 SMPS_Controller2P2ZUpdate(&PMC_2p2z, &dab.PLoop.Feedback,
235 dab.PLoop.Reference, &dab.PLoop.Output);
236
237 }
238
239 // Execute the Current Loop Control
240 if(dab.ILoop.Enable == true)
241 {
242 DPD_TP31_SetHigh();
243 //Bit-shift value used to perform input value normalization
244 dab.ILoop.Feedback = dab.Data.ISecAverageRectified << 3;
245 //adaptive gain factor
246 IMC_2p2z.KfactorCoeffsB = dab.ILoop.AgcFactor;
247 //refresh limits
248 IMC_2p2z.maxOutput = 0x7FFF;
249
250 // Mixing stage from voltage loop 10KHz
251 uint32_t RefBuf = (uint32_t)(dab.ILoop.Reference) *
252 (uint32_t)(dab.VLoop.Output & 0x7FFF);
253 uint16_t ILoopReference = (uint16_t)(RefBuf >> 12);
254
255 // Mixing stage from power loop 10KHz
256 RefBuf = (uint32_t)ILoopReference * (uint32_t)(dab.PLoop.Output & 0x7FFF);
257 ILoopReference = (int16_t)(RefBuf >> 15);
258
259 XFT_SMPS_Controller2P2ZUpdate(&IMC_2p2z, &dab.ILoop.Feedback,
260 ILoopReference, &dab.ILoop.Output);
261
262 // Control loop output copied to control phase
263 dab.Pwm.ControlPhase = (((uint32_t)(dab.Pwm.ControlDutyCycle) *
264 (uint32_t)dab.ILoop.Output) >> 15); //range 0..180
265
266 dab.Pwm.ControlPhase += dab.Pwm.DeadTimeLow;
267
268 // Clamping value of control phase
269 if(dab.Pwm.ControlPhase > (dab.Pwm.ControlDutyCycle - MIN_PHASE_SHIFTED_PULSE))
270 dab.Pwm.ControlPhase = dab.Pwm.ControlDutyCycle - MIN_PHASE_SHIFTED_PULSE;
271 // Clamping value of control phase
272 else if(dab.Pwm.ControlPhase < dab.Pwm.DeadTimeLow)
273 dab.Pwm.ControlPhase = dab.Pwm.DeadTimeLow;
274
275 DPD_TP31_SetLow();
276 }
277}

288static void Dev_PwrCtrl_AdaptiveGainUpdate(void)
289{
290
291 // Calculate the primary voltage in terms of Volts
292 uint16_t DAB_PrimaryVoltage = __builtin_divud((VprimAveraging.AverageValue * VPRI_SCALER), VPRI_FACTOR);
293
294 if(dab.PowerDirection == PWR_CTRL_CHARGING)
295 {
296 // Apply AGC when primary voltage is greater than the minimum VIN AGC threshold
297 if(DAB_PrimaryVoltage > AGC_MINIMUM_VIN_THRESHOLD)
298 dab.ILoop.AgcFactor = (int16_t) (0x7FFF &
299 __builtin_divud(AGC_DAB_FACTOR, DAB_PrimaryVoltage));
300
301 else // AGC is not active
302 dab.ILoop.AgcFactor = 0x7FFF;
303 }
304
305 // Reserved for future Development
306 if(dab.PowerDirection == PWR_CTRL_DISCHARGING)
307 {
308 }
309}

321void Dev_PwrCtrl_PrimToSecPHDegree(void)
322{
323 // Calculation of Primary and Secondary degrees phase
324 // Normalize phase to 0..90.
325 uint32_t buff = ((unsigned long)dab.Pwm.ControlPhase) << DEGREES_PHASE_SCALER;
326 uint16_t buf = __builtin_divud( buff ,dab.Pwm.ControlDutyCycle);
327 buff = __builtin_muluu(buf, PRI_TO_SEC_PHASE_DEGREES_LIMIT);
328 buf = __builtin_divud( buff ,DEGREES_PHASE_SCALING_10);
329
330 // Calculation result for phase value scaled by 10x
331 dab.Pwm.ControlPhase_P2S_Degreex10 = buf;
332
333 // Clamping value for degrees phase when it exceeds the 90.0 degrees
334 if(dab.Pwm.ControlPhase_P2S_Degreex10 >= 900)
335 {
336 // Override the calculated control phase value
337 dab.Pwm.ControlPhase_P2S_Degreex10 = 899;
338 }
339}

349void Dev_PwrCtrl_DeadTimeAdjust(void)
350{
351 uint16_t NewDT = 0;
352
353 // Deadtime values adjusted on rough phase range. experimentally measured values.
354 // If the control phase is less than 29.0 degrees phase
355 if (dab.Pwm.ControlPhase_P2S_Degreex10 < 290)
356 {NewDT = 2000;} // Equivalent to 500ns DeadTime
357 // If the control phase is within the range of 33.0 and 80.0 degrees phase
358 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 330)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 800))
359 {NewDT = 1500;} // Equivalent to 375ns DeadTime
360 // If the control phase is within the range of 80.5 and 87.5 degrees phase
361 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 805)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 875))
362 {NewDT = 700;} // Equivalent to 175ns DeadTime
363 // If the control phase is greater than 88.0 degrees phase
364 else if (dab.Pwm.ControlPhase_P2S_Degreex10 > 880)
365 {NewDT = 600;} // Equivalent to 150ns DeadTime
366
367 // When new dead-time did not satisfy any condition in degrees phase
368 //retain the previous dead-time value
369 if(NewDT)
370 {
371 // Minimum clamping value of dead-time
372 if (NewDT < MINIMUM_DEADTIME)
373 NewDT = MINIMUM_DEADTIME;
374
375 // Write the new dead-time to the PWM
376 dab.Pwm.DeadTimeLow = NewDT;
377 dab.Pwm.DeadTimeHigh = NewDT;
378
379 // Clear NewDT variable
380 NewDT=0;
381 }
382}

392void Dev_PwrCtrl_PeriodModulator(void)
393{
394 static uint16_t decimPM;
395 decimPM++;
396
397 if(dab.Pwm.ControlPhase_P2S_Target > PRI_TO_SEC_PHASE_TARGET)
398 dab.Pwm.ControlPhase_P2S_Target = PRI_TO_SEC_PHASE_TARGET;//clamp cutoff while modulating period
399 if(dab.Pwm.ControlPhase_P2S_Target == 0)
400 dab.Pwm.ControlPhase_P2S_Target = PRI_TO_SEC_PHASE_TARGET;
401
402 if(decimPM>=8)
403 {
404 if (dab.Pwm.ControlPhase_P2S_Degreex10 < dab.Pwm.ControlPhase_P2S_Target-5)
405 {
406 if((dab.Pwm.ControlPeriod > MIN_PWM_PERIOD) && (dab.Pwm.LowPowerSlowMode == 0))
407 {
408 dab.Pwm.ControlPeriod-= PERIODSTEP;
409 }
410 else
411 {
412 // When phase shift between primary to secondary is 68 degrees,
413 // DAB runs in low power mode
414 if (dab.Pwm.ControlPhase_P2S_Degreex10 < 680)
415 {
416 dab.Pwm.LowPowerSlowMode =1;
417 }
418 }
419 }
420
421 if (dab.Pwm.ControlPhase_P2S_Degreex10 > dab.Pwm.ControlPhase_P2S_Target + 5)
422 {
423 if((dab.Pwm.ControlPeriod < MAX_PWM_PERIOD) && (dab.Pwm.ControlPhase_P2S_Degreex10 > 20))
424 {
425 dab.Pwm.ControlPeriod+= PERIODSTEP;
426 }
427 }
428
429 if(dab.Pwm.LowPowerSlowMode == 1)
430 {
431 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 600)
432 {
433 dab.Pwm.LowPowerSlowMode = 0;
434 }
435 else
436 {
437 if(dab.Pwm.ControlPeriod < MIN_PWM_PERIOD)
438 {
439 dab.Pwm.ControlPeriod += PERIODSTEP;
440 }
441 else
442 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 330 )//snap out
443 {
444 dab.Pwm.LowPowerSlowMode = 0;
445 }
446 }
447 }
448 decimPM=0;
449 }
