78void PBV_Change_from_Sixteen_to_Eight(uint16_t * sixteenPtr, uint8_t * eightPtr, uint16_t length)
79{
80 for (uint16_t i = 0, j = 0; i < (length * 2); i = i + 2, j++)
81 {
82 eightPtr[i] = (uint8_t) (sixteenPtr[j] >> 8);
83 eightPtr[i + 1] = (uint8_t) (sixteenPtr[j] & 0x00FF);
84 }
85}

97void PBV_Change_from_Eight_to_Sixteen(uint8_t * eightPtr, uint16_t * sixteenPtr, uint16_t length)
98{
99 uint8_t length_odd = (uint8_t) (length & 0x01);
100 uint16_t length_t;
101
102 uint16_t i, j;
103
104 if (length_odd)
105 {
106 length_t = (length & 0xFFFE);
107 }
108 else
109 {
110 length_t = length;
111 }
112
113 for (i = 0, j = 0; i < length_t; i = i + 2, j++)
114 {
115 sixteenPtr[j] = ((uint16_t) (eightPtr[i] << 8)) + eightPtr[i + 1];
116 }
117 if (length_odd)
118 {
119 sixteenPtr[++j] = ((uint16_t) (eightPtr[length-1] << 8));
120 }
121
122}

174{

193void App_Send_To_PBV(PBV_Datatype_TX_t * ptr)
194{
195 if (ptr->PBV_Message_State == PBV_MESSAGE_INIT
196 ||
197 ptr->PBV_Message_State == PBV_MESSAGE_TRANSMITTED
198 ||
199 ptr->PBV_Message_State == PBV_MESSAGE_TX_ERROR // TODO: for CAN. If the tx fails in arbitration. Introduce a RETRY mechanism.
200 )
201 {
202 appPbvFuncs.linkDataTX(ptr);
203 ptr->PBV_Message_State = PBV_MESSAGE_TRANSMIT_START;
204 }
205}

232static void App_PBV_Task(void)
233{
234 if (appPbvToSystemPtr->PBV_Message_State == PBV_MESSAGE_READY_TO_RECEIVE
235 ||
236 appPbvToSystemPtr->PBV_Message_State == PBV_STATE_RECEIVING
237 )
238 {
239 appPbvToSystemPtr->PBV_Message_State = appPbvFuncs.rxGui();
240 }
241
242 if (appSystemToPbvPtr->PBV_Message_State == PBV_MESSAGE_TRANSMIT_START
243 ||
244 appSystemToPbvPtr->PBV_Message_State == PBV_MESSAGE_TRANSMITTING
245 )
246 {
247 appSystemToPbvPtr->PBV_Message_State = appPbvFuncs.txGui();
248 }
249
250 if (appSystemToPbvAsciiPtr->PBV_Message_State == PBV_MESSAGE_TRANSMIT_START
251 ||
252 appSystemToPbvAsciiPtr->PBV_Message_State == PBV_MESSAGE_TRANSMITTING
253 )
254 {
255 appSystemToPbvAsciiPtr->PBV_Message_State = appPbvFuncs.txGuiAscii();
256 }
257}
