73void PBV_Change_from_Sixteen_to_Eight(uint16_t * sixteenPtr, uint8_t * eightPtr, uint16_t length)
74{
75 for (uint16_t i = 0, j = 0; i < (length * 2); i = i + 2, j++)
76 {
77 eightPtr[i] = (uint8_t) (sixteenPtr[j] >> 8);
78 eightPtr[i + 1] = (uint8_t) (sixteenPtr[j] & 0x00FF);
79 }
80}

90void PBV_Change_from_Eight_to_Sixteen(uint8_t * eightPtr, uint16_t * sixteenPtr, uint16_t length)
91{
92 uint8_t length_odd = (uint8_t) (length & 0x01);
93 uint16_t length_t;
94
95 uint16_t i, j;
96
97 if (length_odd)
98 {
99 length_t = (length & 0xFFFE);
100 }
101 else
102 {
103 length_t = length;
104 }
105
106 for (i = 0, j = 0; i < length_t; i = i + 2, j++)
107 {
108 sixteenPtr[j] = ((uint16_t) (eightPtr[i] << 8)) + eightPtr[i + 1];
109 }
110 if (length_odd)
111 {
112 sixteenPtr[++j] = ((uint16_t) (eightPtr[length-1] << 8));
113 }
114
115}

180void App_Send_To_PBV(PBV_Datatype_TX_t * ptr)
181{
182 if (ptr->PBV_Message_State == PBV_MESSAGE_INIT
183 ||
184 ptr->PBV_Message_State == PBV_MESSAGE_TRANSMITTED
185 ||
186 ptr->PBV_Message_State == PBV_MESSAGE_TX_ERROR // TODO: for CAN. If the tx fails in arbitration. Introduce a RETRY mechanism.
187 )
188 {
189 appPbvFuncs.linkDataTX(ptr);
190 ptr->PBV_Message_State = PBV_MESSAGE_TRANSMIT_START;
191 }
192}

217static void App_PBV_Task(void)
218{
219 if (appPbvToSystemPtr->PBV_Message_State == PBV_MESSAGE_READY_TO_RECEIVE
220 ||
221 appPbvToSystemPtr->PBV_Message_State == PBV_STATE_RECEIVING
222 )
223 {
224 appPbvToSystemPtr->PBV_Message_State = appPbvFuncs.rxGui();
225 }
226
227 if (appSystemToPbvPtr->PBV_Message_State == PBV_MESSAGE_TRANSMIT_START
228 ||
229 appSystemToPbvPtr->PBV_Message_State == PBV_MESSAGE_TRANSMITTING
230 )
231 {
232 appSystemToPbvPtr->PBV_Message_State = appPbvFuncs.txGui();
233 }
234
235 if (appSystemToPbvAsciiPtr->PBV_Message_State == PBV_MESSAGE_TRANSMIT_START
236 ||
237 appSystemToPbvAsciiPtr->PBV_Message_State == PBV_MESSAGE_TRANSMITTING
238 )
239 {
240 appSystemToPbvAsciiPtr->PBV_Message_State = appPbvFuncs.txGuiAscii();
241 }
242}
