86void PwrCtrl_UpdateADConverterData (void)
87{
88 // Enable the ADC sampling
89 ADC1_SoftwareTriggerEnable();
90
91 if(ADC1_IsConversionComplete(ISEC_CT))
92 dab.Data.ISenseSecondary = ADC1_ConversionResultGet(ISEC_CT);
93
94 if(ADC1_IsConversionComplete(IPRI_CT))
95 dab.Data.ISensePrimary = ADC1_ConversionResultGet(IPRI_CT);
96
97 if(ADC1_IsConversionComplete(VRAIL_5V))
98 dab.Data.VRail_5V = ADC1_ConversionResultGet(VRAIL_5V);
99
100 if(ADC1_IsConversionComplete(TEMP))
101 dab.Data.Temperature = ADC1_ConversionResultGet(TEMP);
102
103 if(ADC1_IsConversionComplete(VPRI)){
104 dab.Data.VPriVoltage = ADC1_ConversionResultGet(VPRI);
105 VprimAveraging.Accumulator += dab.Data.VPriVoltage;
106 VprimAveraging.Counter = VprimAveraging.Counter + 1;
107 }
108
109 if(ADC1_IsConversionComplete(VSEC)){
110 dab.Data.VSecVoltage = ADC1_ConversionResultGet(VSEC);
111 VsecAveraging.Accumulator += dab.Data.VSecVoltage;
112 VsecAveraging.Counter = VsecAveraging.Counter + 1;
113 }
114
115 if(ADC1_IsConversionComplete(ISEC_AVG)){
116 dab.Data.ISecAverage = ADC1_ConversionResultGet(ISEC_AVG);
117 dab.Data.ISecAverageRectified = abs(dab.Data.ISecAverage - dab.Data.ISecSensorOffset);
118 IsecAveraging.Accumulator += dab.Data.ISecAverageRectified;
119 IsecAveraging.Counter = IsecAveraging.Counter + 1;
120 }
121}

133void PwrCtrl_10KHzVPLoopPrepareData(void)
134{
135 static uint16_t cnt = 0;
136
137 //Interleave the execution of VLoop and PLoop control for 10KHz execution
138 if(++cnt == (VPLOOP_ILOOP_EXE_RATIO))
139 {
140 // Averaging of Secondary Voltage
141 VsecAveraging.AverageValue = (uint16_t)(__builtin_divud(VsecAveraging.Accumulator, VsecAveraging.Counter));
142 VsecAveraging.Accumulator = 0;
143 VsecAveraging.Counter = 0;
144
145 #if(OPEN_LOOP_PBV == false)
146 //Condition for control loop execution controlling the loop enable bit
147 if((dab.VLoop.Enable == false) && (VLoopInterleaveExec == true))
148 #endif
149 {
150 // Enable Vloop control
151 dab.VLoop.Enable = true;
152 dab.PLoop.Enable = false;
153
154 // Averaging of Primary Voltage
155 VprimAveraging.AverageValue = (uint16_t)(__builtin_divud(VprimAveraging.Accumulator, VprimAveraging.Counter));
156 VprimAveraging.Accumulator = 0;
157 VprimAveraging.Counter = 0;
158
159 // Compute the Adaptive Gain
160 PwrCtrl_AdaptiveGainUpdate();
161 }
162
163 #if(OPEN_LOOP_PBV == false)
164 else if((dab.PLoop.Enable == false) && (VLoopInterleaveExec == false))
165 #endif
166 {
167 // Enable PLoop control
168 dab.VLoop.Enable = false;
169 dab.PLoop.Enable = true;
170
171 // Averaging of Secondary Current
172 IsecAveraging.AverageValue = (uint16_t)(__builtin_divud(IsecAveraging.Accumulator, IsecAveraging.Counter));
173 IsecAveraging.Accumulator = 0;
174 IsecAveraging.Counter = 0;
175
176 // Bit-shift value used to perform input value normalization
177 // Scaled the feedback to Power (Watts in units)
178 uint32_t buf = (uint32_t)IsecAveraging.AverageValue *
179 (uint32_t)VsecAveraging.AverageValue * POWER_FACTOR;
180
181 // scale back the 14 bit from the POWER_RESOLUTION calculation
182 // to get the Watts value for Power Loop
183 buf >>= POWER_SCALER;
184
185 // Transfer to SecPower data member the power computation in [Watts]
186 dab.Data.SecPower = buf;
187
188 }
189 cnt = 0;
190 }
191}

204void PwrCtrl_ControlLoopExecute(void)
205{
206 // Execute the Voltage Loop Control
207 if((dab.VLoop.Enable == true) && (VLoopInterleaveExec == true))
208 {
209 VLoopInterleaveExec = false;
210
211 if(dab.PowerDirection == PWR_CTRL_CHARGING)
212 {
213 VMC_2p2z.KfactorCoeffsB = 0x7FFF;
214 VMC_2p2z.maxOutput = 0x7FFF;
215
216 // Bit-shift value used to perform input value normalization
217 dab.VLoop.Feedback = VsecAveraging.AverageValue << 3;
218 dab.VLoop.Reference = dab.VLoop.Reference << 3;
219
220 // Execute the Voltage Loop Control
221 SMPS_Controller2P2ZUpdate(&VMC_2p2z, &dab.VLoop.Feedback,
222 dab.VLoop.Reference, &dab.VLoop.Output);
223
224 // Reset the Vloop reference to its original scaling
225 dab.VLoop.Reference = dab.VLoop.Reference >> 3;
226 }
227 }
228
229 // Execute the Power Loop Control
230 if((dab.PLoop.Enable == true) && (VLoopInterleaveExec == false))
231 {
232 VLoopInterleaveExec = true;
233
234 dab.PLoop.Feedback = dab.Data.SecPower;
235 dab.PLoop.Reference = dab.PLoop.Reference;
236
237 // Execute the Power Loop Control
238 SMPS_Controller2P2ZUpdate(&PMC_2p2z, &dab.PLoop.Feedback,
239 dab.PLoop.Reference, &dab.PLoop.Output);
240
241 }
242
243 // Execute the Current Loop Control
244 if(dab.ILoop.Enable == true)
245 {
246 DPD_TP31_SetHigh();
247 //Bit-shift value used to perform input value normalization
248 dab.ILoop.Feedback = dab.Data.ISecAverageRectified << 3;
249 //adaptive gain factor
250 IMC_2p2z.KfactorCoeffsB = dab.ILoop.AgcFactor;
251 //refresh limits
252 IMC_2p2z.maxOutput = 0x7FFF;
253
254 // Mixing stage from voltage loop 10KHz
255 uint32_t RefBuf = (uint32_t)(dab.ILoop.Reference) *
256 (uint32_t)(dab.VLoop.Output & 0x7FFF);
257 uint16_t ILoopReference = (uint16_t)(RefBuf >> 12);
258
259 // Mixing stage from power loop 10KHz
260 RefBuf = (uint32_t)ILoopReference * (uint32_t)(dab.PLoop.Output & 0x7FFF);
261 ILoopReference = (int16_t)(RefBuf >> 15);
262
263 XFT_SMPS_Controller2P2ZUpdate(&IMC_2p2z, &dab.ILoop.Feedback,
264 ILoopReference, &dab.ILoop.Output);
265
266 // Control loop output copied to control phase
267 dab.Pwm.ControlPhase = (((uint32_t)(dab.Pwm.ControlDutyCycle) *
268 (uint32_t)dab.ILoop.Output) >> 15); //range 0..180
269
270 dab.Pwm.ControlPhase += dab.Pwm.DeadTimeLow;
271
272 // Clamping value of control phase
273 if(dab.Pwm.ControlPhase > (dab.Pwm.ControlDutyCycle - MIN_PHASE_SHIFTED_PULSE))
274 dab.Pwm.ControlPhase = dab.Pwm.ControlDutyCycle - MIN_PHASE_SHIFTED_PULSE;
275 // Clamping value of control phase
276 else if(dab.Pwm.ControlPhase < dab.Pwm.DeadTimeLow)
277 dab.Pwm.ControlPhase = dab.Pwm.DeadTimeLow;
278
279 DPD_TP31_SetLow();
280 }
281}

292static void PwrCtrl_AdaptiveGainUpdate(void)
293{
294
295 // Calculate the primary voltage in terms of Volts
296 uint16_t DAB_PrimaryVoltage = __builtin_divud((VprimAveraging.AverageValue * VPRI_SCALER), VPRI_FACTOR);
297
298 if(dab.PowerDirection == PWR_CTRL_CHARGING)
299 {
300 // Apply AGC when primary voltage is greater than the minimum VIN AGC threshold
301 if(DAB_PrimaryVoltage > AGC_MINIMUM_VIN_THRESHOLD)
302 dab.ILoop.AgcFactor = (int16_t) (0x7FFF &
303 __builtin_divud(AGC_DAB_FACTOR, DAB_PrimaryVoltage));
304
305 else // AGC is not active
306 dab.ILoop.AgcFactor = 0x7FFF;
307 }
308
309 // Reserved for future Development
310 if(dab.PowerDirection == PWR_CTRL_DISCHARGING)
311 {
312 }
313}

325void PwrCtrl_PrimToSecPHDegree(void)
326{
327 // Calculation of Primary and Secondary degrees phase
328 // Normalize phase to 0..90.
329 uint32_t buff = ((unsigned long)dab.Pwm.ControlPhase) << DEGREES_PHASE_SCALER;
330 uint16_t buf = __builtin_divud( buff ,dab.Pwm.ControlDutyCycle);
331 buff = __builtin_muluu(buf, PRI_TO_SEC_PHASE_DEGREES_LIMIT);
332 buf = __builtin_divud( buff ,DEGREES_PHASE_SCALING_10);
333
334 // Calculation result for phase value scaled by 10x
335 dab.Pwm.ControlPhase_P2S_Degreex10 = buf;
336
337 // Clamping value for degrees phase when it exceeds the 90.0 degrees
338 if(dab.Pwm.ControlPhase_P2S_Degreex10 >= 900)
339 {
340 // Override the calculated control phase value
341 dab.Pwm.ControlPhase_P2S_Degreex10 = 899;
342 }
343}

353void PwrCtrl_DeadTimeAdjust(void)
354{
355 uint16_t NewDT = 0;
356
357 // Deadtime values adjusted on rough phase range. experimentally measured values.
358 // If the control phase is less than 29.0 degrees phase
359 if (dab.Pwm.ControlPhase_P2S_Degreex10 < 290)
360 {NewDT = 2000;} // Equivalent to 500ns DeadTime
361 // If the control phase is within the range of 33.0 and 80.0 degrees phase
362 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 330)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 800))
363 {NewDT = 1500;} // Equivalent to 375ns DeadTime
364 // If the control phase is within the range of 80.5 and 87.5 degrees phase
365 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 805)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 875))
366 {NewDT = 700;} // Equivalent to 175ns DeadTime
367 // If the control phase is greater than 88.0 degrees phase
368 else if (dab.Pwm.ControlPhase_P2S_Degreex10 > 880)
369 {NewDT = 600;} // Equivalent to 150ns DeadTime
370
371 // When new dead-time did not satisfy any condition in degrees phase
372 //retain the previous dead-time value
373 if(NewDT)
374 {
375 // Minimum clamping value of dead-time
376 if (NewDT < MINIMUM_DEADTIME)
377 NewDT = MINIMUM_DEADTIME;
378
379 // Write the new dead-time to the PWM
380 dab.Pwm.DeadTimeLow = NewDT;
381 dab.Pwm.DeadTimeHigh = NewDT;
382
383 // Clear NewDT variable
384 NewDT=0;
385 }

396void PwrCtrl_PeriodModulator(void)
397{
398 static uint16_t decimPM;
399 decimPM++;
400
401 if(dab.Pwm.ControlPhase_P2S_Target > PRI_TO_SEC_PHASE_TARGET)
402 dab.Pwm.ControlPhase_P2S_Target = PRI_TO_SEC_PHASE_TARGET;//clamp cutoff while modulating period
403 if(dab.Pwm.ControlPhase_P2S_Target == 0)
404 dab.Pwm.ControlPhase_P2S_Target = PRI_TO_SEC_PHASE_TARGET;
405
406 if(decimPM>=8)
407 {
408 if (dab.Pwm.ControlPhase_P2S_Degreex10 < dab.Pwm.ControlPhase_P2S_Target-5)
409 {
410 if((dab.Pwm.ControlPeriod > MIN_PWM_PERIOD) && (dab.Pwm.LowPowerSlowMode == 0))
411 {
412 dab.Pwm.ControlPeriod-= PERIODSTEP;
413 }
414 else
415 {
416 // When phase shift between primary to secondary is 68 degrees,
417 // DAB runs in low power mode
418 if (dab.Pwm.ControlPhase_P2S_Degreex10 < 680)
419 {
420 dab.Pwm.LowPowerSlowMode =1;
421 }
422 }
423 }
424
425 if (dab.Pwm.ControlPhase_P2S_Degreex10 > dab.Pwm.ControlPhase_P2S_Target + 5)
426 {
427 if((dab.Pwm.ControlPeriod < MAX_PWM_PERIOD) && (dab.Pwm.ControlPhase_P2S_Degreex10 > 20))
428 {
429 dab.Pwm.ControlPeriod+= PERIODSTEP;
430 }
431 }
432
433 if(dab.Pwm.LowPowerSlowMode == 1)
434 {
435 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 600)
436 {
437 dab.Pwm.LowPowerSlowMode = 0;
438 }
439 else
440 {
441 if(dab.Pwm.ControlPeriod < MIN_PWM_PERIOD)
442 {
443 dab.Pwm.ControlPeriod += PERIODSTEP;
444 }
445 else
446 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 330 )//snap out
447 {
448 dab.Pwm.LowPowerSlowMode = 0;
449 }
450 }
451 }
452 decimPM=0;
453 }
