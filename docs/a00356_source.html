147bool FAULT_CheckMax(FAULT_OBJ_T *faultInput, int16_t faultSource, FAULT_CALLBACK callback)
148{
149
150 if ((faultSource >= faultInput->Val1_Threshold) && (faultInput->FaultActive == false))
151 {
152 faultInput->FaultCounter++;
153
154 if (faultInput->FaultCounter > faultInput->CounterSet)
155 {
156 faultInput->FaultLatch = true;
157
158 if (callback != NULL)
159 {
160 callback(); // Call Back function used to quickly shutdown system if needed when fault detected
161 }
162
163 faultInput->FaultActive = true;
164 faultInput->FaultCounter = 0U;
165 }
166 }
167 else if ((faultSource < (faultInput->Val1_Hysteresis)) && (faultInput->FaultActive == true))
168 {
169 faultInput->FaultCounter++;
170 if (faultInput->FaultCounter > faultInput->CounterReset)
171 {
172 faultInput->FaultCounter = 0U;
173 faultInput->FaultActive = false;
174 }
175 }
176 else
177 {
178 faultInput->FaultCounter = 0U;
179 }
180
181 return (faultInput->FaultActive);
182
183}

308bool FAULT_CheckRange(FAULT_OBJ_T *faultInput, int16_t inputValue, FAULT_CALLBACK callback)
309{
310 if (faultInput->FaultActive == false)
311 {
312 if ((inputValue <= faultInput->Val1_Threshold) ||
313 (inputValue >= faultInput->Val2_Threshold))
314 {
315 if (++faultInput->FaultCounter >= faultInput->CounterSet)
316 {
317 faultInput->FaultLatch = true;
318
319 if (callback != NULL)
320 {
321 callback(); // Call Back function used to quickly shutdown system if needed when fault detected
322 }
323 faultInput->FaultActive = true;
324 faultInput->FaultCounter = 0U;
325 }
326 }
327 else
328 {
329 faultInput->FaultCounter = 0U;
330 }
331 }
332 else
333 {
334 if ((inputValue > faultInput->Val1_Hysteresis) &&
335 (inputValue < faultInput->Val2_Hysteresis))
336 {
337 if (++faultInput->FaultCounter > faultInput->CounterReset)
338 {
339 faultInput->FaultCounter = 0U;
340 faultInput->FaultActive = false;
341 }
342 }
343 else
344 {
345 faultInput->FaultCounter = 0U;
346 }
347 }
348 return (faultInput->FaultActive);
349}
