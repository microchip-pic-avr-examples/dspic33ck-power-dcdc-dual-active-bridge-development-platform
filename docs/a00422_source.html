74void ControlLoop_Interrupt_CallBack(void)
75{
76 GPIO_1_SetHigh();
77
78 // Update the ADC data member
79 Dev_PwrCtrl_UpdateConverterData();
80
81 // Execute the fault detection
82 Dev_Fault_Execute();
83
84 #if(OPEN_LOOP_PBV == false)
85 // Execute Power Converter Control Loop
86 Dev_PwrCtrl_ControlLoopExecute();
87 #endif
88
89 #if(DAC_DEBUG == true)
90 // This is for DAC debugging purposes
91// CMP2_DACDataWrite(dab.VLoop.Reference);
92// CMP2_DACDataWrite(dab.ILoop.Reference << 2);
93 CMP2_DACDataWrite(dab.PLoop.Reference >> 2);
94 #endif
95
96 #if (true == DPDB_TEST_RUN)
97
98 // Connect TP11 -> Pot1 & TP12 -> Pot2 in digital Power Development Board
99 uint16_t Pot1 = dab.Data.ISecAverage;
100 uint16_t Pot2 = dab.Data.ISenseSecondary;
101
102 // Calculate the Frequency based on the Potentiometer 1 voltage
103 dab.Pwm.ControlPeriod = (uint16_t)(MIN_PWM_PERIOD + (Pot1 * ADC_PERIOD_RANGE));
104
105 // Change the control Phase with Potentiometer 2
106 dab.Pwm.ControlPhase = (uint16_t)(Pot2 * ADC_SCALER * (dab.Pwm.ControlPeriod >> 1));
107
108 #endif
109
110 // Update PWM Properties
111 Dev_PwrCtrl_PWM_Update(&dab);
112
113 GPIO_1_SetLow();
114
115}

126static void Dev_PwrCtrl_UpdateConverterData (void)
127{
128 ADC1_SoftwareTriggerEnable();
129
130 dab.Data.ISecAverage = ADC1_ConversionResultGet(ISEC_AVG);
131 dab.Data.ISecAverageRectified = abs(dab.Data.ISecAverage - dab.Data.ISecSensorOffset);
132
133 IsecAveraging.Accumulator += dab.Data.ISecAverageRectified;
134 IsecAveraging.Counter = IsecAveraging.Counter + 1;
135
136 dab.Data.VSecVoltage = ADC1_ConversionResultGet(VSEC);
137 VsecAveraging.Accumulator += dab.Data.VSecVoltage;
138 VsecAveraging.Counter = VsecAveraging.Counter + 1;
139
140 dab.Data.ISenseSecondary = ADC1_ConversionResultGet(ISEC_CT);
141 dab.Data.ISensePrimary = ADC1_ConversionResultGet(IPRI_CT);
142
143 dab.Data.VPriVoltage = ADC1_ConversionResultGet(VPRI);
144
145 dab.Data.VRail_5V = ADC1_ConversionResultGet(VRAIL_5V);
146 dab.Data.Temperature = ADC1_ConversionResultGet(TEMP);
147
148}

161static void Dev_PwrCtrl_ControlLoopExecute(void)
162{
163 static uint16_t cnt = 0;
164 static bool VLoopExec = true;
165
166 //Interleave the execution of VLoop and PLoop control for 10KHz execution
167 if(++cnt == 5)
168 {
169 // Averaging of Secondary Voltage
170 VsecAveraging.AverageValue = (uint16_t)(__builtin_divud(VsecAveraging.Accumulator, VsecAveraging.Counter));
171 VsecAveraging.Accumulator = 0;
172 VsecAveraging.Counter = 0;
173
174 //Condition for control loop execution controlling the loop enable bit
175 if((dab.VLoop.Enable == false) && (VLoopExec == true)){
176 dab.VLoop.Enable = true;
177 dab.PLoop.Enable = false;
178
179 }
180 else if((dab.PLoop.Enable == false) && (VLoopExec == false)){
181 dab.VLoop.Enable = false;
182 dab.PLoop.Enable = true;
183
184 // Averaging of Secondary Current
185 IsecAveraging.AverageValue = (uint16_t)(__builtin_divud(IsecAveraging.Accumulator, IsecAveraging.Counter));
186 IsecAveraging.Accumulator = 0;
187 IsecAveraging.Counter = 0;
188 }
189 cnt = 0;
190 }
191
192 // Execute the Voltage Loop Control
193 if((dab.VLoop.Enable == true) && (VLoopExec == true))
194 {
195
196 VLoopExec = false;
197
198 if(dab.PowerDirection == PWR_CTRL_CHARGING)
199 {
200 VMC_2p2z.KfactorCoeffsB = 0x7FFF;
201 VMC_2p2z.maxOutput = 0x7FFF;
202
203 // Bit-shift value used to perform input value normalization
204 dab.VLoop.Feedback = VsecAveraging.AverageValue << 4;
205 dab.VLoop.Reference = dab.VLoop.Reference << 4;
206
207 // Execute the Voltage Loop Control
208 SMPS_Controller2P2ZUpdate(&VMC_2p2z, &dab.VLoop.Feedback,
209 dab.VLoop.Reference, &dab.VLoop.Output);
210
211 // Reset the Vloop reference to its original scaling
212 dab.VLoop.Reference = dab.VLoop.Reference >> 4;
213 }
214 }
215
216 // Execute the Power Loop Control
217 if((dab.PLoop.Enable == true) && (VLoopExec == false))
218 {
219 VLoopExec = true;
220
221 // Bit-shift value used to perform input value normalization
222 // Scaled the feedback to Power (Watts in units)
223 uint32_t buf = (uint32_t)IsecAveraging.AverageValue *
224 (uint32_t)VsecAveraging.AverageValue * POWER_RESOLUTION;
225 // scale back the 14 bit from the POWER_RESOLUTION calculation
226 // to get the Watts value for Power Loop
227 buf >>=14;
228
229 dab.Data.SecPower = buf;
230
231 dab.PLoop.Feedback = dab.Data.SecPower;
232 dab.PLoop.Reference = dab.PLoop.Reference;
233
234 // Execute the Power Loop Control
235 SMPS_Controller2P2ZUpdate(&PMC_2p2z, &dab.PLoop.Feedback,
236 dab.PLoop.Reference, &dab.PLoop.Output);
237
238 }
239
240 // Execute the Current Loop Control
241 if(dab.ILoop.Enable == true)
242 {
243 //Bit-shift value used to perform input value normalization
244 dab.ILoop.Feedback = dab.Data.ISecAverageRectified << 3;
245 //adaptive gain factor
246 IMC_2p2z.KfactorCoeffsB = 0x7FFF;//dab.ILoop.AgcFactor;
247 //refresh limits
248 IMC_2p2z.maxOutput = 0x7FFF;
249
250 // Mixing stage from voltage loop 10KHz
251 uint32_t RefBuf = (uint32_t)(dab.ILoop.Reference) *
252 (uint32_t)(dab.VLoop.Output & 0x7FFF);
253 uint16_t ILoopReference = (uint16_t)(RefBuf>>12);
254
255 // Mixing stage from power loop 10KHz
256 RefBuf = (uint32_t)ILoopReference * (uint32_t)(dab.PLoop.Output & 0x7FFF);
257 ILoopReference = (int16_t)(RefBuf >> 15);
258
259 XFT_SMPS_Controller2P2ZUpdate(&IMC_2p2z, &dab.ILoop.Feedback,
260 ILoopReference, &dab.ILoop.Output);
261
262 // Control loop output copied to control phase
263 dab.Pwm.ControlPhase = (((uint32_t)(dab.Pwm.ControlDutyCycle) *
264 (uint32_t)dab.ILoop.Output) >> 15); //range 0..180
265 }
