46uint16_t Dev_PwrCtrl_UpdateAverage(AVERAGING_t* data, uint16_t sample)

67bool Dev_PwrCtrl_RampReference(START_UP_RAMP_t* rampUp)
68{
69 rampUp->RampComplete = false;
70
71 if (++rampUp->Counter > rampUp->Delay)
72 {
73 rampUp->Counter = 0;
74
75 if(*rampUp->ptrReference == *rampUp->ptrReferenceTarget)
76 {
77 rampUp->RampComplete = true;
78 }
79 else if ((*rampUp->ptrReference + rampUp->StepSize) < *rampUp->ptrReferenceTarget)
80 {
81 *rampUp->ptrReference += rampUp->StepSize;
82 }
83 else if ((*rampUp->ptrReference - rampUp->StepSize) > *rampUp->ptrReferenceTarget)
84 {
85 *rampUp->ptrReference -= rampUp->StepSize;
86 }
87 else
88 {
89 *rampUp->ptrReference = *rampUp->ptrReferenceTarget;
90 rampUp->RampComplete = true;
91 }
92 }
93
94 return (rampUp->RampComplete);
95}
