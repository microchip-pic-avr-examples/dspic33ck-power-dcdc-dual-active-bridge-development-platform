139static uint16_t Average_Temp_ADC_Samples(void) {
140 if (devTempData.BufferFull)

153static uint16_t Temp_Calculate_Average(uint16_t * buffer, uint16_t length) {
154 uint16_t index = 0;
155 uint32_t sum = 0;
156 for (index = 0; index <length; index++)
157 sum +=buffer[index];
158 return (sum >> MAX_NUM_SAMPLES_TEMP_BUFFER_AS_EXP_OF_TWO);
159}
