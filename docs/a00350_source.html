62void Fault_Handler(void)
63{
64 // Drive the fault pin to Low when Fault trips
65 FAULT_SetLow();
66
67 // Turn off PWM output
68 PwrCtrl_PWM_Disable();
69
70 // set the fault active bit
71 dab.Status.bits.FaultActive = 1;
72
73 // clear the running bit
74 dab.Status.bits.Running = 0;
75
76}

86void Fault_Initialize(void)
87{
88 // Initialize Primary Over Current Protection
89 FAULT_Init(&dab.Fault.Object.IPrimaryOCP, IPRI_OC_THRES_TRIG,
90 IPRI_OC_THRES_CLEAR, IPRI_OC_T_BLANK_TRIG, IPRI_OC_T_BLANK_CLEAR);
91
92 // Initialize Secondary Over Current Protection
93 FAULT_Init(&dab.Fault.Object.ISecondaryOCP, ISEC_OC_THRES_TRIG,
94 ISEC_OC_THRES_CLEAR, ISEC_OC_T_BLANK_TRIG, ISEC_OC_T_BLANK_CLEAR);
95
96 // Initialize Primary Over Voltage Protection
97 FAULT_Init(&dab.Fault.Object.VPrimaryOVP, VPRI_OV_THRES_TRIG,
98 VPRI_OV_THRES_CLEAR, VPRI_OV_T_BLANK_TRIG, VPRI_OV_T_BLANK_CLEAR);
99
100 // Initialize Primary Over Voltage Protection
101 FAULT_Init(&dab.Fault.Object.VPrimaryUVP, VPRI_OV_THRES_TRIG,
102 VPRI_OV_THRES_CLEAR, VPRI_OV_T_BLANK_TRIG, VPRI_OV_T_BLANK_CLEAR);
103
104 // Initialize Secondary Over Voltage Protection
105 FAULT_Init(&dab.Fault.Object.VSecondaryOVP, VSEC_OV_THRES_TRIG,
106 VSEC_OV_THRES_CLEAR, VSEC_OV_T_BLANK_TRIG, VSEC_OV_T_BLANK_CLEAR);
107
108 // Initialize Secondary Over Voltage Protection
109 FAULT_Init(&dab.Fault.Object.VSecondaryUVP, VSEC_OV_THRES_TRIG,
110 VSEC_OV_THRES_CLEAR, VSEC_OV_T_BLANK_TRIG, VSEC_OV_T_BLANK_CLEAR);
111
112 // Initialize 5V Rail instability Protection
113 FAULT_Init(&dab.Fault.Object.VRail_5V, VRAIL_5V_UV_THRES_TRIG,
114 VRAIL_5V_UV_THRES_CLEAR, VRAIL_5V_UV_T_BLANK_TRIG, VRAIL_5V_UV_T_BLANK_CLEAR);
115
116 // Initialize Over Temperature Protection
117 FAULT_Init(&dab.Fault.Object.PowerSupplyOTP, MAX_TEMPERATURE_THRESHOLD_RAW,
118 OVER_TEMP_UPPER_THRESHOLD_WITH_HYST,FAULT_PERSISTENCE_COUNT_TEMP, FAULT_PERSISTENCE_COUNT_TEMP);
119
120#if (FAULT_SHORT_CCT == true)
121 // Initialize short circuit fault protection with comparators
122 Fault_EnableShortCircuitProtection();
123#endif
124
125}

136void Fault_Execute(void)
137{
138 // secondary over current fault handler
139 #if (FAULT_ISEC_OC)
140 FAULT_CheckMax(&dab.Fault.Object.ISecondaryOCP, dab.Data.ISenseSecondary, &Fault_Handler);
141 #endif
142
143 // secondary over voltage fault handler
144 #if (FAULT_VSEC_OV)
145 FAULT_CheckMax(&dab.Fault.Object.VSecondaryOVP, dab.Data.VSecVoltage, &Fault_Handler);
146 #endif
147
148 // primary over current fault handler
149 #if(FAULT_IPRI_OC)
150 FAULT_CheckMax(&dab.Fault.Object.IPrimaryOCP, dab.Data.ISensePrimary, &Fault_Handler);
151 #endif
152
153 // primary over voltage fault handler
154 #if (FAULT_VPRI_OV)
155 FAULT_CheckMax(&dab.Fault.Object.VPrimaryOVP, dab.Data.VPriVoltage, &Fault_Handler);
156 #endif
157
158 // primary over voltage fault handler
159 #if (FAULT_VPRI_UV)
160 FAULT_CheckMin(&dab.Fault.Object.VPrimaryUVP, dab.Data.VPriVoltage, &Fault_Handler);
161 #endif
162
163 // primary over voltage fault handler
164 #if (FAULT_VRAIL_5V)
165 FAULT_CheckMin(&dab.Fault.Object.VRail_5V, dab.Data.VRail_5V, &Fault_Handler);
166 #endif
167
168 // Hardware short circuit
169 if(CMP1_StatusGet() || CMP3_StatusGet()){
170
171 Fault_Handler();
172
173 // Set fault bits
174 dab.Fault.Object.ISenseSCP.FaultActive = 1;
175 dab.Fault.Object.ISenseSCP.FaultLatch = 1;
176
177 }
178
179 #if(LOAD_DISCONNECT)
180 // Protection when Load is removed by accident.
181 //DAB does not sink power in this modulation. Voltage builds up on output.
182 if(dab.PowerDirection==PWR_CTRL_CHARGING)
183 {
184 if((iSecAveraging.AverageValue <= ISEC_LOAD_STEP_CLAMP>>2) &&
185 (vSecAveraging.AverageValue > (dab.VLoop.Reference + VSEC_LOAD_STEP_CLAMP)) &&
186 (dab.Properties.IReference >= 1) )
187 {
188 loadDisconnect = true;
189 Fault_Handler();
190 }
191 }
192 if(dab.PowerDirection==PWR_CTRL_DISCHARGING)
193 {
194 if((iSecAveraging.AverageValue <= ISEC_LOAD_STEP_CLAMP>>2) &&
195 (vPrimAveraging.AverageValue > (dab.VLoop.Reference + VPRIM_LOAD_STEP_CLAMP)) &&
196 (dab.Properties.IReference >= 1) )
197 {
198 loadDisconnect = true;
199 Fault_Handler();
200 }
201 }
202 #endif
203
204 // Identify the fault that trips
205 dab.Fault.FaultDetected = Fault_GetFlags();
206}

217void Fault_Reset(void)
218{
219 // Drive the fault pin to high to allow PWM signal drive
220 FAULT_SetHigh();
221
222 // Clear fault Objects FaultActive bit
223 dab.Fault.Object.VPrimaryOVP.FaultActive = 0;
224 dab.Fault.Object.VSecondaryOVP.FaultActive = 0;
225 dab.Fault.Object.IPrimaryOCP.FaultActive = 0;
226 dab.Fault.Object.ISecondaryOCP.FaultActive = 0;
227 dab.Fault.Object.ISenseSCP.FaultActive = 0;
228 dab.Fault.Object.PowerSupplyOTP.FaultActive = 0;
229 dab.Fault.Object.VRail_5V.FaultActive = 0;
230
231 // Clear fault Objects FaultLatch bit
232 dab.Fault.Object.VPrimaryOVP.FaultLatch = 0;
233 dab.Fault.Object.VSecondaryOVP.FaultLatch = 0;
234 dab.Fault.Object.IPrimaryOCP.FaultLatch = 0;
235 dab.Fault.Object.ISecondaryOCP.FaultLatch = 0;
236 dab.Fault.Object.ISenseSCP.FaultLatch = 0;
237 dab.Fault.Object.VRail_5V.FaultLatch = 0;
238 dab.Fault.Object.PowerSupplyOTP.FaultActive = 0;
239
240 loadDisconnect = false;
241
242}

253static void Fault_EnableShortCircuitProtection(void)
254{
255 // on dsPIC33CK DP-PIM:
256 // CMP1B used for short circuit protection on the secondary side
257 // CMP3B used for short circuit protection on the primary side
258
259 // initialize thresholds of comparators used for short circuit protection
260 CMP_ISEC_SC_DACDataWrite(ISEC_SC_THRES_TRIG);
261 CMP_IPRI_SC_DACDataWrite(IPRI_SC_THRES_TRIG);
262
263 // turn on comparator DACs
264 CMP_IPRI_SC_DACEnable();
265 CMP_ISEC_SC_DACEnable();
266
267}
