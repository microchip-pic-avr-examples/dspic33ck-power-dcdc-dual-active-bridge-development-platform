81/* Data type for the 3-pole 3-zero (3P3Z) controller
82
83 Summary:
84 Data type for the 3-pole 3-zero (3P3Z) controller
85
86 Description:
87 The 3P3Z controller is the digital implementation of the analog type III
88 controller. This is a filter which generates a compensator characteristic
89 considering three poles and three zeros. This controller requires four
90 feedback errors multiplied by their associated coefficients plus the three
91 latest controller output values multiplied by their associated coefficients
92 along the delay line to provide proper compensation of the power converter.
93 The coefficients are determined externally using simulation tools.
94
95 The SMPS_3P3Z_T data structure contains a pointer to derived
96 coefficients in X-space and pointer to error/control history samples in
97 Y-space. User must declare variables for the derived coefficients and
98 the error history samples.
99
100 The abCoefficients referenced by the SMPS_3P3Z_T data structure are
101 derived from the coefficients B0-B3 plus A1-A3. These will be declared
102 in external arrays. The SMPS_3P3Z_T data structure just holds pointers
103 to these arrays.
104
105 The coefficients will be determined by simulation tools, their outputs are
106 given as floating point numbers. These numbers will be copied into the
107 declared arrays after they have been converted into 16-bit integer numbers.
108
109*/
110
111typedef struct
112{
113 int16_t* aCoefficients ; // Pointer to A coefficients located in X-space
114 int16_t* bCoefficients ; // Pointer to B coefficients located in X-space
115 int16_t* controlHistory ; // Pointer to 3 delay-line samples located in Y-space
116 // with the first sample being the most recent
117 int16_t* errorHistory ; // Pointer to 4 delay-line samples located in Y-space
118 // with the first sample being the most recent
119 uint16_t preShift ; // Normalization from ADC-resolution to Q15 (R/W)
120 int16_t postShift ; // Normalization bit-shift from Q15 to PWM register resolution (R/W)
121 int16_t postScaler ; // Controller output post-scaler (R/W)
122 uint16_t minOutput ; // Minimum output value used for clamping (R/W)
123 uint16_t maxOutput ; // Maximum output value used for clamping (R/W)
124} SMPS_3P3Z_T;

127/* Data type for the 2-pole 2-zero (2P2Z) controller
128
129 Summary:
130 Data type for the 2-pole 2-zero (2P2Z) controller
131
132 Description:
133 The 2P2Z controller is the digital implementation of the analog type II
134 controller. This is a filter which generates a compensator characteristic
135 considering two poles and two zeros. This controller requires three feedback
136 error multiplied by their associated coefficients plus the two latest
137 controller output values multiplied by their associated coefficients along
138 the delay line to provide proper compensation of the power converter.
139 The coefficients are determined externally using simulation tools.
140
141 The SMPS_2P2Z_T data structure contains a pointer to derived
142 coefficients in X-space and pointer to error/control history samples in
143 Y-space. User must declare variables for the derived coefficients and
144 the error history samples.
145
146 The abCoefficients referenced by the SMPS_2P2Z_T data structure are
147 derived from the coefficients B0-B2 plus A1-A2. These will be declared
148 in external arrays. The SMPS_2P2Z_T data structure and just holds
149 pointers to these arrays.
150
151 The coefficients will be determined by simulation tools, which output
152 is given as floating point numbers. These numbers will be copied into
153 the declared arrays after they have been converted into 16-bit integer
154 numbers.
155*/
156typedef struct
157{
158 int16_t* aCoefficients ; // Pointer to A coefficients located in X-space
159 int16_t* bCoefficients ; // Pointer to B coefficients located in X-space
160 int16_t* controlHistory ; // Pointer to 2 delay-line samples located in Y-space
161 // with the first sample being the most recent
162 int16_t* errorHistory ; // Pointer to 3 delay-line samples located in Y-space
163 // with the first sample being the most recent
164 uint16_t preShift ; // Normalization from ADC-resolution to Q15 (R/W)
165 int16_t postShift ; // Normalization bit-shift from Q15 to PWM register resolution (R/W)
166 int16_t postScaler ; // Controller output post-scaler (R/W)
167 uint16_t minOutput ; // Minimum output value used for clamping (R/W)
168 uint16_t maxOutput ; // Maximum output value used for clamping (R/W)

171} SMPS_2P2Z_T ;

174/* Data type for the PID controller
175
176 Summary:
177 Data type for the PID controller
178
179 Description:
180 Data type for the Proportional Integral Derivative (PID) controller
181
182 This digital implementation of a PID controller is a filter which
183 generates a compensator characteristic considering the values of the
184 coefficients KA, KB, KC these coefficients will determine the
185 converter's frequency response. These coefficients are determined
186 externally using simulation tools.
187
188 This function call includes the pointer to the controller data
189 structure, pointer of the input source register, control reference
190 value, and to the pointer to the output register.
191 */
192typedef struct {
193
194 int16_t* abcCoefficients; // Pointer to A, B & C coefficients located in X-space
195 // These coefficients are derived from
196 // the PID gain values - Kp, Ki and Kd
197 int16_t* errorHistory; // Pointer to 3 delay-line samples located in Y-space
198 // with the first sample being the most recent
199 int16_t controlHistory; // Stores the most recent controller output (n-1)
200 int16_t postScaler; // PID basic Coefficient scaling Factor
201 int16_t preShift ; // Normalization from ADC-resolution to Q15 (R/W)
202 int16_t postShift; // Normalization from DSP to PWM register
203 uint16_t minOutput; // Minimum output value used for clamping
204 uint16_t maxOutput; // Maximum output value used for clamping
205
206} SMPS_PID_T;
