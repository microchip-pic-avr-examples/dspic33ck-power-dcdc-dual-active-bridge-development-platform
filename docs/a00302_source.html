117int8_t Dev_Temp_Get_Temperature_Celcius(void){
118
119 uint8_t index = 0;
120
121 TEMP_NTC_LUT_t point0;
122
123 devTempData.AdcAverage = Average_Temp_ADC_Samples();
124
125 if (devTempData.AdcAverage == 0) return 0;
126
127 if (devTempData.AdcAverage > 4056)
128 return NTC_lookuptable[0].temperature - 40;
129
130 if (devTempData.AdcAverage < 379)
131 return NTC_lookuptable[33].temperature - 40;
132
133 for (index = 0; index < 32; index++) {
134 if (devTempData.AdcAverage > NTC_lookuptable[index].ADC_Val) {
135 point0 = NTC_lookuptable[index];
136 break;
137 }
138 }
139 point0.temperatureCelsius=point0.temperature-40;
140 return point0.temperatureCelsius;
141}

163void Dev_Temp_Get_ADC_Sample(void)
164{
165 // making sure that array does not go out of bounds.
166 if (devTempData.BufferIndex >= (MAX_NUM_SAMPLES_TEMP_BUFFER - 1))
167 {
168 devTempData.BufferIndex = 0;
169 devTempData.BufferFull = 1;
170 }
171
172 // use API from power controller to read temperature
173 devTempData.AdcReading = Dev_PwrCtrl_GetAdc_Temperature();
174 devTempData.TempBuffer[devTempData.BufferIndex++] = devTempData.AdcReading;
175
176}

185static uint16_t Average_Temp_ADC_Samples(void) {
186 if (devTempData.BufferFull)
187 return Temp_Calculate_Average(devTempData.TempBuffer, MAX_NUM_SAMPLES_TEMP_BUFFER);
188 else
189 return 0;
190}

199static uint16_t Temp_Calculate_Average(uint16_t * buffer, uint16_t length) {
200 uint16_t index = 0;
201 uint32_t sum = 0;
202 for (index = 0; index <length; index++)
203 sum +=buffer[index];
204 return (sum >> MAX_NUM_SAMPLES_TEMP_BUFFER_AS_EXP_OF_TWO);
205}
