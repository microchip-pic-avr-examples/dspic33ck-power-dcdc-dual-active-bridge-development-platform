437static void PwrCtrl_AdaptiveGainUpdate(void)
438{
439
440 if(dab.PowerDirection == PWR_CTRL_CHARGING)
441 {
442 // Calculate the primary voltage in terms of Volts
443 uint16_t DAB_PrimaryVoltage = __builtin_divud((vPrimAveraging.AverageValue * VPRI_SCALER), VPRI_FACTOR);
444
445 // Apply AGC when primary voltage is greater than the minimum VIN AGC threshold
446 if(DAB_PrimaryVoltage > AGC_MINIMUM_VIN_THRESHOLD)
447 dab.ILoop.AgcFactor = (int16_t) (0x7FFF &
448 __builtin_divud(AGC_VOLTAGE_FACTOR, DAB_PrimaryVoltage));
449 else // AGC is not active
450 dab.ILoop.AgcFactor = 0x7FFF;
451
452
453 #if(ENABLE_VLOOP_AGC == true)
454 if(dab.ILoop.Reference > AGC_MINIMUM_CURRENT_THRESHOLD)
455 dab.VLoop.AgcFactor = (int16_t) (0x7FFF &
456 __builtin_divud(AGC_CURRENT_FACTOR, dab.ILoop.Reference));
457 else // AGC is not active
458 dab.VLoop.AgcFactor = 0x7FFF;
459 #endif
460 }
461
462 // Reserved for future Development
463 if(dab.PowerDirection == PWR_CTRL_DISCHARGING)
464 {
465
466 // Calculate the secondary voltage in terms of Volts
467 uint16_t DAB_SecondaryVoltage = __builtin_divud((vSecAveraging.AverageValue * VSEC_SCALER), VSEC_FACTOR);
468
469 // Apply AGC when secondary voltage is greater than the minimum VIN AGC threshold
470 if(DAB_SecondaryVoltage > AGC_MINIMUM_VIN_THRESHOLD_SEC)
471 dab.ILoop.AgcFactor = (int16_t) (0x7FFF &
472 __builtin_divud(AGC_VOLTAGE_FACTOR_SEC, DAB_SecondaryVoltage));
473 else // AGC is not active
474 dab.ILoop.AgcFactor = 0x7FFF;
475
476 }
477}

489void PwrCtrl_PrimToSecPHDegree(void)
490{
491 // Calculation of Primary and Secondary degrees phase
492 // Normalize phase to 0..90.
493 uint32_t buff = ((unsigned long)dab.Pwm.ControlPhase) << DEGREES_PHASE_SCALER;
494 uint16_t buf = __builtin_divud( buff ,dab.Pwm.ControlDutyCycle);
495 buff = __builtin_muluu(buf, PRI_TO_SEC_PHASE_DEGREES_LIMIT);
496 buf = __builtin_divud( buff ,DEGREES_PHASE_SCALING_10);
497
498 // Calculation result for phase value scaled by 10x
499 dab.Pwm.ControlPhase_P2S_Degreex10 = buf;
500
501 // Clamping value for degrees phase when it exceeds the 90.0 degrees
502 if(dab.Pwm.ControlPhase_P2S_Degreex10 >= 900)
503 {
504 // Override the calculated control phase value
505 dab.Pwm.ControlPhase_P2S_Degreex10 = 899;
506 }

517void PwrCtrl_DeadTimeAdjust(void)
518{
519 uint16_t NewDT = 0;
520
521 // Deadtime values adjusted on rough phase range. experimentally measured values.
522 // If the control phase is less than 29.0 degrees phase
523 if (dab.Pwm.ControlPhase_P2S_Degreex10 < 290)
524 {NewDT = 2000;} // Equivalent to 500ns DeadTime
525 // If the control phase is within the range of 33.0 and 80.0 degrees phase
526 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 330)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 800))
527 {NewDT = 1500;} // Equivalent to 375ns DeadTime
528 // If the control phase is within the range of 80.5 and 87.5 degrees phase
529 else if ((dab.Pwm.ControlPhase_P2S_Degreex10 > 805)&&(dab.Pwm.ControlPhase_P2S_Degreex10 < 875))
530 {NewDT = 700;} // Equivalent to 175ns DeadTime
531 // If the control phase is greater than 88.0 degrees phase
532 else if (dab.Pwm.ControlPhase_P2S_Degreex10 > 880)
533 {NewDT = 600;} // Equivalent to 150ns DeadTime
534
535 // When new dead-time did not satisfy any condition in degrees phase
536 //retain the previous dead-time value
537 if(NewDT)
538 {
539 // Minimum clamping value of dead-time
540 if (NewDT < MINIMUM_DEADTIME)
541 NewDT = MINIMUM_DEADTIME;
542
543 // Write the new dead-time to the PWM
544 dab.Pwm.DeadTimeLow = NewDT;
545 dab.Pwm.DeadTimeHigh = NewDT;
546
547 // Clear NewDT variable
548 NewDT=0;
549 }

579 {

583 {
584 dab.Pwm.LowPowerSlowMode =1;

586 }

590 {

592 {
593 dab.Pwm.ControlPeriod+= PERIODSTEP;
594 }
595 }
596

598 {
599 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 660)

602 }

608 }

610 if (dab.Pwm.ControlPhase_P2S_Degreex10 > 440 )//snap out
611 {
612 dab.Pwm.LowPowerSlowMode = 0;
613 }
