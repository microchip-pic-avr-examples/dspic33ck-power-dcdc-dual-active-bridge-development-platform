61void Dev_Fault_Handler(void)
62{
63 // Drive the fault pin to Low when Fault trips
64 FAULT_SetLow();
65
66 // Turn off PWM output
67 Dev_PwrCtrl_PWM_Disable();
68
69 // set the fault active bit
70 dab.Status.bits.FaultActive = 1;
71
72 // clear the running bit
73 dab.Status.bits.Running = 0;
74
75}

86void Dev_Fault_Initialize(void)
87{
88 // Initialize Primary Over Current Protection
89 FAULT_Init(&dab.Fault.Object.IPrimaryOCP, IPRI_OC_THRES_TRIG,
90 IPRI_OC_THRES_CLEAR, IPRI_OC_T_BLANK_TRIG, IPRI_OC_T_BLANK_CLEAR);
91
92 // Initialize Secondary Over Current Protection
93 FAULT_Init(&dab.Fault.Object.ISecondaryOCP, ISEC_OC_THRES_TRIG,
94 ISEC_OC_THRES_CLEAR, ISEC_OC_T_BLANK_TRIG, ISEC_OC_T_BLANK_CLEAR);
95
96 // Initialize Primary Over Voltage Protection
97 FAULT_Init(&dab.Fault.Object.VPrimaryOVP, VPRI_OV_THRES_TRIG,
98 VPRI_OV_THRES_CLEAR, VPRI_OV_T_BLANK_TRIG, VPRI_OV_T_BLANK_CLEAR);
99
100 // Initialize Secondary Over Voltage Protection
101 FAULT_Init(&dab.Fault.Object.VSecondaryOVP, VSEC_OV_THRES_TRIG,
102 VSEC_OV_THRES_CLEAR, VSEC_OV_T_BLANK_TRIG, VSEC_OV_T_BLANK_CLEAR);
103
104 // Initialize 5V Rail instability Protection
105 FAULT_Init(&dab.Fault.Object.VRail_5V, VRAIL_5V_UV_THRES_TRIG,
106 VRAIL_5V_UV_THRES_CLEAR, VRAIL_5V_UV_T_BLANK_TRIG, VRAIL_5V_UV_T_BLANK_CLEAR);
107
108 // Initialize Over Temperature Protection
109 FAULT_Init(&dab.Fault.Object.PowerSupplyOTP, MAX_TEMPERATURE_THRESHOLD_RAW,
110 OVER_TEMP_UPPER_THRESHOLD_WITH_HYST,FAULT_PERSISTENCE_COUNT_TEMP, FAULT_PERSISTENCE_COUNT_TEMP);
111
112#if (FAULT_SHORT_CCT == true)
113 // Initialize short circuit fault protection with comparators
114 Dev_Fault_EnableShortCircuitProtection();
115#endif
116 // Hardware Fault Initialize
117 CMP1_EventCallbackRegister(&Dev_Fault_Handler);
118 CMP3_EventCallbackRegister(&Dev_Fault_Handler);
119
120}

131void Dev_Fault_Execute(void)
132{
133 // secondary over current fault handler
134 #if (FAULT_ISEC_OC)
135 FAULT_CheckMax(&dab.Fault.Object.ISecondaryOCP, dab.Data.ISenseSecondary, &Dev_Fault_Handler);
136 #endif
137
138 // secondary over voltage fault handler
139 #if (FAULT_VSEC_OV)
140 FAULT_CheckMax(&dab.Fault.Object.VSecondaryOVP, dab.Data.VSecVoltage, &Dev_Fault_Handler);
141 #endif
142
143 // primary over current fault handler
144 #if(FAULT_IPRI_OC)
145 FAULT_CheckMax(&dab.Fault.Object.IPrimaryOCP, dab.Data.ISensePrimary, &Dev_Fault_Handler);
146 #endif
147
148 // primary over voltage fault handler
149 #if (FAULT_VPRI_OV)
150 FAULT_CheckMax(&dab.Fault.Object.VPrimaryOVP, dab.Data.VPriVoltage, &Dev_Fault_Handler);
151 #endif
152
153 // primary over voltage fault handler
154 #if (FAULT_VRAIL_5V)
155 FAULT_CheckMin(&dab.Fault.Object.VRail_5V, dab.Data.VRail_5V, &Dev_Fault_Handler);
156 #endif
157
158 // Hardware short circuit
159 if(CMP1_StatusGet() || CMP3_StatusGet()){
160 dab.Fault.Object.ISenseSCP.FaultActive = 1;
161 dab.Fault.Object.ISenseSCP.FaultLatch = 1;
162 dab.Status.bits.FaultActive = 1;
163 }
164
165 // Identify the fault that trips
166 dab.Fault.FaultDetected = Dev_Fault_GetFlags();
167}

178void Dev_Fault_Reset(void)
179{
180 // Drive the fault pin to high to allow PWM signal drive
181 FAULT_SetHigh();
182
183 // Clear fault Objects FaultActive bit
184 dab.Fault.Object.VPrimaryOVP.FaultActive = 0;
185 dab.Fault.Object.VSecondaryOVP.FaultActive = 0;
186 dab.Fault.Object.IPrimaryOCP.FaultActive = 0;
187 dab.Fault.Object.ISecondaryOCP.FaultActive = 0;
188 dab.Fault.Object.ISenseSCP.FaultActive = 0;
189 dab.Fault.Object.PowerSupplyOTP.FaultActive = 0;
190 dab.Fault.Object.VRail_5V.FaultActive = 0;
191
192 // Clear fault Objects FaultLatch bit
193 dab.Fault.Object.VPrimaryOVP.FaultLatch = 0;
194 dab.Fault.Object.VSecondaryOVP.FaultLatch = 0;
195 dab.Fault.Object.IPrimaryOCP.FaultLatch = 0;
196 dab.Fault.Object.ISecondaryOCP.FaultLatch = 0;
197 dab.Fault.Object.ISenseSCP.FaultLatch = 0;
198 dab.Fault.Object.VRail_5V.FaultLatch = 0;
199 dab.Fault.Object.PowerSupplyOTP.FaultActive = 0;
200
201}

212static void Dev_Fault_EnableShortCircuitProtection(void)
213{
214 // on dsPIC33CK DP-PIM:
215 // CMP1B used for short circuit protection on the secondary side
216 // CMP3B used for short circuit protection on the primary side
217
218 // initialize thresholds of comparators used for short circuit protection
219 CMP_ISEC_SC_DACDataWrite(ISEC_SC_THRES_TRIG);
220 CMP_IPRI_SC_DACDataWrite(IPRI_SC_THRES_TRIG);
221
222 // turn on comparator DACs
223 CMP_IPRI_SC_DACEnable();
224 CMP_ISEC_SC_DACEnable();
225
226}
