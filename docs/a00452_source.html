95/*
96 * ************************************************************************************************
97 * Summary:
98 * Configures interrupt trigger source setting of an individual analog input
99 *
100 * Parameters:
101 * - inputIndex: index of channel (0 for AN0, 1 for AN1, etc)
102 * - triggerSource: trigger source
103 *
104 * Returns:
105 * 0: failure
106 * 1: success
107 *
108 * Description:
109 * This function configures the trigger source selection for corresponding analog inputs by
110 * extracting the user setting from the function parameter "triggerSource"
111 *
112 * ***********************************************************************************************/
113static __inline__ uint16_t ADC_ADInput_SetTriggerSource(uint16_t inputIndex, ADTRIG_TRGSRC_t triggerSource)
114{
115 uint16_t fres = 1;
116 uint8_t *regptr;
117
118 // Check if given ADC input index is within available range
119 if (inputIndex >= ADC_ANINPUT_COUNT) return(0);
120
121 // Determine register set offset
122 regptr = (uint8_t *)&ADTRIG0L; // get register block base address
123 regptr += (uint8_t)inputIndex; // add offset
124 *regptr = (uint8_t)triggerSource;
125
126 // Check if WRITE operation was successful
127 fres &= (bool)((*regptr == (uint8_t)triggerSource));
128
129 return(1);
130}
