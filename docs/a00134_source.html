63void CLC2_Initialize(void)
64{
65 // Set the CLC2 to the options selected in the User Interface
66
67 // LCEN enabled; LCINTN disabled; LCPOL inverted; LCOUT Low; LCINTP disabled; LCMODE OR-XOR; LCOE enabled;
68 CLC2CONL = 0x80A1;
69 // LCG1POL non_inverted; LCG2POL non_inverted; LCG3POL non_inverted; LCG4POL non_inverted;
70 CLC2CONH = 0x0;
71 // LCD2S CLCINB I/O pin; LCD1S CLCINA I/O pin; LCD4S PWM Event A; LCD3S CLCINC I/O pin;
72 CLC2SELL = 0x0;
73 // LCG1D1T disabled; LCG1D3T disabled; LCG2D2T disabled; LCG1D2T disabled; LCG2D1T disabled; LCG2D4T disabled; LCG1D4T disabled; LCG2D3T disabled; LCG1D1N disabled; LCG1D3N disabled; LCG2D2N disabled; LCG1D2N disabled; LCG2D1N disabled; LCG2D4N disabled; LCG1D4N disabled; LCG2D3N disabled;
74 CLC2GLSL = 0x0;
75 // LCG3D1T disabled; LCG3D2T disabled; LCG4D1T disabled; LCG3D3T disabled; LCG4D2T disabled; LCG3D4T disabled; LCG4D3T disabled; LCG4D4T enabled; LCG3D1N disabled; LCG3D2N disabled; LCG4D1N disabled; LCG3D3N disabled; LCG4D2N disabled; LCG3D4N disabled; LCG4D3N disabled; LCG4D4N disabled;
76 CLC2GLSH = 0x8000;
77
78 CLC2_PositiveEdgeCallbackRegister(&CLC2_PositiveEdgeCallback);
79 CLC2_NegativeEdgeCallbackRegister(&CLC2_NegativeEdgeCallback);
80
81 CLC2_Enable();
82
83}

124void CLC2_Tasks( void )
125{
126 if(IFS7bits.CLC2PIF == 1)
127 {
128 if(NULL != CLC2_PositiveEdgeHandler)
129 {
130 (*CLC2_PositiveEdgeHandler)();
131 }
132
133 // Clear the CLC2 Positive Edge Interrupt flag
134 IFS7bits.CLC2PIF = 0;
135 }
136 if(IFS11bits.CLC2NIF == 1)
137 {
138 if(NULL != CLC2_NegativeEdgeHandler)
139 {
140 (*CLC2_NegativeEdgeHandler)();
141 }
142
143 // Clear the CLC2 Negative Edge Interrupt flag
144 IFS11bits.CLC2NIF = 0;
145 }
146}
