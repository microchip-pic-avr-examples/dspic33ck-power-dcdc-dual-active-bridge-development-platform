45void PwrCtrl_Initialize(void)
46{
47 // Clear dab references
48 dab.Properties.VPriReference = 0;
49 dab.Properties.VSecReference = 0;
50 dab.Properties.IReference = 0;
51 dab.Properties.PwrReference = 0;
52
53 // Initialize the Control Period and Control Phase during start-up
54 dab.Pwm.ControlPeriod = MAX_PWM_PERIOD;
55 dab.Pwm.DeadTimeHigh = MINIMUM_DEADTIME;
56 dab.Pwm.DeadTimeLow = MINIMUM_DEADTIME;
57 dab.Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
58
59 // Initialize the DAB to charging state
60 dab.PowerDirection = PWR_CTRL_CHARGING;
61
62 // Initialize Start-Up ramp settings
63 PwrCtrl_StartUpInitialize();
64
65 // Initialize Power Control Loop
66 PwrCtrl_ControlLoopInitialize();
67
68 // Disable PWM output by setting the PWM override bits to High
69 PwrCtrl_PWM_Disable();
70
71 // Enable PWM peripheral
72 PWM_Enable();
73
74 // Update the Period, Duty Cycle and Phases of the PWMs based on
75 // the given Control period and Control Phase
76 PwrCtrl_PWM_Update(&dab);
77}

102void PwrCtrl_Reset(void)
103{
104 //set the period to maximum
105 dab.Pwm.ControlPeriod = MAX_PWM_PERIOD;
106 dab.Pwm.ControlPhase = dab.Pwm.DeadTimeLow;
107 dab.Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
108 dab.Pwm.PBVControlPhaseTarget = dab.Pwm.DeadTimeLow;
109
110 // Reset the power control references
111 dab.Properties.VPriReference = 0;
112 dab.Properties.VSecReference = 0;
113 dab.Properties.IReference = 0;
114 dab.Properties.PwrReference = 0;
115
116 // Initialize current loop reference to 0, to be controlled externally
117 dab.ILoop.Reference = 0;
118 // Initialize power loop reference to 0, to be controlled externally
119 dab.PLoop.Reference = 0;
120 // Initialize voltage loop reference to current secondary voltage
121 dab.VLoop.Reference = 0;
122
123 // Set the AGC to 1
124 dab.VLoop.AgcFactor = 0x7FFF;
125 dab.ILoop.AgcFactor = 0x7FFF;
126 dab.PLoop.AgcFactor = 0x7FFF;
127
128 // Reset Control Loop Histories
129 PwrCtrl_ResetControlLoopHistories();
130}

186static void PwrCtrl_StartUpInitialize(void)
187{
188 // Initialize Voltage ramp-up settings
189 dab.VRamp.ptrReference = (uint16_t*)&dab.VLoop.Reference;
190 dab.VRamp.ptrReferenceTarget = &dab.Properties.VSecReference;
191 dab.VRamp.StepSize = 1;
192 dab.VRamp.Delay = 20;
193 dab.VRamp.Counter = 0;
194 dab.VRamp.RampComplete = 0;
195
196 //Initialize Current ramp-up settings
197 dab.IRamp.ptrReference = (uint16_t*)&dab.ILoop.Reference;
198 dab.IRamp.ptrReferenceTarget = (uint16_t*)&dab.Properties.IReference;
199 dab.IRamp.StepSize = 1;
200 dab.IRamp.Delay = 60;
201 dab.IRamp.Counter = 0;
202 dab.IRamp.RampComplete = 0;
203
204 //Initialize Power ramp-up settings
205 dab.PRamp.ptrReference = (uint16_t*)&dab.PLoop.Reference;
206 dab.PRamp.ptrReferenceTarget = &dab.Properties.PwrReference;
207 dab.PRamp.StepSize = 1;
208 dab.PRamp.Delay = 4;
209 dab.PRamp.Counter = 0;
210 dab.PRamp.RampComplete = 0;
211
212 #if defined (OPEN_LOOP_PBV) && (OPEN_LOOP_PBV == true)
213 // The PWM Period bits [2:0] needs to be mask when using cascaded PWM setup
214 // (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
215 uint16_t PeriodMask = 0x7;
216
217 // Initialize Voltage ramp-up settings for Period control
218 dab.Pwm.ControlPeriod = dab.Pwm.ControlPeriod & ~(PeriodMask);
219 dab.Pwm.PBVPeriodTarget = dab.Pwm.PBVPeriodTarget & ~(PeriodMask);
220
221 dab.VRamp.ptrReference = &dab.Pwm.ControlPeriod;
222 dab.VRamp.ptrReferenceTarget = &dab.Pwm.PBVPeriodTarget;
223 dab.VRamp.StepSize = 0xE;
224 dab.VRamp.Delay = 0;
225
226 //Initialize Current ramp-up settings for Phase control
227 dab.Pwm.ControlPhase = dab.Pwm.ControlPhase & ~(PeriodMask);
228 dab.Pwm.PBVControlPhaseTarget = dab.Pwm.PBVControlPhaseTarget & ~(PeriodMask);
229
230 dab.IRamp.ptrReference = &dab.Pwm.ControlPhase;
231 dab.IRamp.ptrReferenceTarget = &dab.Pwm.PBVControlPhaseTarget;
232 dab.IRamp.StepSize = 0xE;
233 dab.IRamp.Delay = 0;
234
235#endif
236}
