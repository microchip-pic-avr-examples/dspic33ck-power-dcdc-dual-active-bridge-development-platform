65void Dev_PwrCtrl_StateMachine(POWER_CONTROL_t* pcInstance)
66{
67 switch (pcInstance->State)
68 {
69 case PWRCTRL_STATE_INITIALIZE:
70 PCS_INIT_handler(pcInstance);
71 break;
72
73 case PWRCTRL_STATE_FAULT_DETECTION:
74 PCS_WAIT_IF_FAULT_ACTIVE_handler(pcInstance);
75 break;
76
77 case PWRCTRL_STATE_STANDBY:
78 PCS_STANDBY_handler(pcInstance);
79 break;
80
81 case PWRCTRL_STATE_SOFT_START:
82 PCS_SOFT_START_handler(pcInstance);
83 break;
84
85 case PWRCTRL_STATE_ONLINE:
86 PCS_UP_AND_RUNNING_handler(pcInstance);
87 break;
88
89 default:
90 pcInstance->State = PWRCTRL_STATE_INITIALIZE;
91 break;
92 }
93}

105static void PCS_INIT_handler(POWER_CONTROL_t* pcInstance)
106{
107 #if (CURRENT_CALIBRATION == true)
108 // Execute current sensor offset calibration
109 Dev_CurrentSensorOffsetCal();
110
111 // Checks if the calibration is complete
112 if (Dev_CurrentSensor_Get_CalibrationStatus())
113 #endif
114 {
115 // Current sensor calibration is complete. Update the offset of the current sensor
116 pcInstance->Data.ISecSensorOffset = Dev_CurrentSensor_Get_Offset();
117
118 // Ensure PWM output is disabled
119 Dev_PwrCtrl_PWM_Disable(pcInstance);
120
121 // Reset fault objects status bits
122 Dev_Fault_Reset();
123
124 // Clear power control fault active bit
125 pcInstance->Status.bits.FaultActive = 0;
126 // Clear power control running bit
127 pcInstance->Status.bits.Running = 0;
128 // Clear power control enable bit
129 pcInstance->Properties.Enable = 0;
130
131 // Next State assigned to STATE_FAULT_DETECTION
132 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
133 }
134}

171static void PCS_STANDBY_handler(POWER_CONTROL_t* pcInstance)
172{
173 // Check for fault event
174 if (pcInstance->Fault.FaultDetected)
175 {
176 // Clear power control enable bit
177 pcInstance->Properties.Enable = 0;
178
179 // State back to STATE_FAULT_DETECTION
180 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
181 }
182
183 // NOTE: Power control enable is controlled externally
184 else if (pcInstance->Properties.Enable)
185 {
186 // Reset fault objects status bits
187 Dev_Fault_Reset();
188
189 // reset the PWM settings in Standby mode
190 pcInstance->Pwm.ControlPeriod = MAX_PWM_PERIOD;
191 pcInstance->Pwm.ControlPhase = 0;
192 pcInstance->Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
193 pcInstance->Pwm.PBVControlPhaseTarget = 0;
194
195 // Enable power control running bit
196 pcInstance->Status.bits.Running = 1;
197
198 // ToDo: Not yet applied; check this again
199 Dev_Fault_ClearHardwareFaults();
200
201 //
202 Dev_PwrCtrl_ControlLoopInitialize();
203
204 // Enable PWM physical output
205 Dev_PwrCtrl_PWM_Enable(pcInstance);
206
207 //Enable current loop control
208 pcInstance->ILoop.Enable = true;
209
210 // Initialize current loop reference to 0, to be controlled externally
211 pcInstance->ILoop.Reference = 0;
212
213 // Initialize power loop reference to 0, to be controlled externally
214 pcInstance->PLoop.Reference = 0;
215
216 // Initialize voltage loop reference to current secondary voltage
217 pcInstance->VLoop.Reference = pcInstance->Data.VSecVoltage;
218
219 // Next State assigned to STATE_SOFT_START
220 pcInstance->State = PWRCTRL_STATE_SOFT_START;
221 }
222}

236static void PCS_SOFT_START_handler(POWER_CONTROL_t* pcInstance)
237{
238 // Check for fault event
239 if (pcInstance->Fault.FaultDetected)
240 {
241 // Clear power control enable bit
242 pcInstance->Properties.Enable = 0;
243
244 // State back to STATE_FAULT_DETECTION
245 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
246 }
247
248 // Check if Enable bit has been cleared
249 else if (!pcInstance->Properties.Enable)
250 {
251 // Disable PWM physical output
252 Dev_PwrCtrl_PWM_Disable(pcInstance);
253
254 // Clear power control running bit
255 pcInstance->Status.bits.Running = 0;
256
257 // State back to STATE_STANDBY
258 pcInstance->State = PWRCTRL_STATE_STANDBY;
259 }
260
261 else
262 {
263 // Ramp Up the Voltage, Current and Power reference
264 Dev_PwrCtrl_RampReference(&pcInstance->VRamp);
265 Dev_PwrCtrl_RampReference(&pcInstance->IRamp);
266 Dev_PwrCtrl_RampReference(&pcInstance->PRamp);
267
268 // Check if ramp up is complete
269 if ((pcInstance->VRamp.RampComplete) && (pcInstance->IRamp.RampComplete)
270 && (pcInstance->PRamp.RampComplete))
271 // Next State assigned to STATE_ONLINE
272 pcInstance->State = PWRCTRL_STATE_ONLINE;
273
274 }
275}

287static void PCS_UP_AND_RUNNING_handler(POWER_CONTROL_t* pcInstance)
288{
289// // Remove after testing LEB
290// PG6IOCONLbits.OVRENH = 0;
291// PG6IOCONLbits.OVRENL = 0;
292// DAC1DATH = 0x7FF;
293// DAC3DATH = 0x7FF;
294//
295 // Check for fault event
296 if (pcInstance->Fault.FaultDetected)
297 {
298 // Clear power control enable bit
299 pcInstance->Properties.Enable = 0;
300
301 // State back to STATE_FAULT_DETECTION
302 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
303 }
304
305 else
306 {
307 // Check if Enable bit has been cleared
308 if (!pcInstance->Properties.Enable)
309 {
310 // Disable PWM physical output
311 Dev_PwrCtrl_PWM_Disable(pcInstance);
312
313 // Clear power control running bit
314 pcInstance->Status.bits.Running = 0;
315
316 // State back to STATE_STANDBY
317 pcInstance->State = PWRCTRL_STATE_STANDBY;
318 }
319
320 #if (OPEN_LOOP_PBV == true)
321 else if ((pcInstance->Pwm.ControlPeriod != pcInstance->Pwm.PBVPeriodTarget) ||
322 (pcInstance->Pwm.ControlPhase != pcInstance->Pwm.PBVControlPhaseTarget))
323 pcInstance->State = PWRCTRL_STATE_SOFT_START;
324 #endif
325
326 // Check if there is change in power control references
327 else if ((pcInstance->ILoop.Reference != pcInstance->Properties.IReference) ||
328 (pcInstance->VLoop.Reference != pcInstance->Properties.VSecReference) ||
329 (pcInstance->PLoop.Reference != pcInstance->Properties.PwrReference))
330
331 // State back to STATE_SOFT_START
332 pcInstance->State = PWRCTRL_STATE_SOFT_START;
333 }
334}
