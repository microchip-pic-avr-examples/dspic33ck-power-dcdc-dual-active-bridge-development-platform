73void Dev_PwrCtrl_StateMachine(POWER_CONTROL_t* pcInstance)
74{
75 switch (pcInstance->State)
76 {
77 case PWRCTRL_STATE_INITIALIZE:
78 PCS_INIT_handler(pcInstance);
79 break;
80
81 case PWRCTRL_STATE_FAULT_DETECTION:
82 PCS_WAIT_IF_FAULT_ACTIVE_handler(pcInstance);
83 break;
84
85 case PWRCTRL_STATE_STANDBY:
86 PCS_STANDBY_handler(pcInstance);
87 break;
88
89 case PWRCTRL_STATE_SOFT_START:
90 PCS_SOFT_START_handler(pcInstance);
91 break;
92
93 case PWRCTRL_STATE_ONLINE:
94 PCS_UP_AND_RUNNING_handler(pcInstance);
95 break;
96
97 default:
98 pcInstance->State = PWRCTRL_STATE_INITIALIZE;
99 break;
100 }
101}

113static void PCS_INIT_handler(POWER_CONTROL_t* pcInstance)
114{
115 #if (CURRENT_CALIBRATION == true)
116 // Execute current sensor offset calibration
117 Dev_CurrentSensorOffsetCal();
118
119 // Checks if the calibration is complete
120 if (Dev_CurrentSensor_Get_CalibrationStatus())
121 #endif
122 {
123 // Current sensor calibration is complete. Update the offset of the current sensor
124 pcInstance->Data.ISecSensorOffset = Dev_CurrentSensor_Get_Offset();
125
126 // Ensure PWM output is disabled
127 Dev_PwrCtrl_PWM_Disable();
128
129 // Reset fault objects status bits
130 Dev_Fault_Reset();
131
132 // Clear power control fault active bit
133 pcInstance->Status.bits.FaultActive = 0;
134 // Clear power control running bit
135 pcInstance->Status.bits.Running = 0;
136 // Clear power control enable bit
137 pcInstance->Properties.Enable = 0;
138
139 // Next State assigned to STATE_FAULT_DETECTION
140 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
141 }
142}

179static void PCS_STANDBY_handler(POWER_CONTROL_t* pcInstance)
180{
181 // Check for fault event
182 if (pcInstance->Fault.FaultDetected)
183 {
184 // Clear power control enable bit
185 pcInstance->Properties.Enable = 0;
186
187 // State back to STATE_FAULT_DETECTION
188 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
189 }
190
191 // NOTE: Power control enable is controlled externally
192 else if (pcInstance->Properties.Enable)
193 {
194 // Reset fault objects status bits
195 Dev_Fault_Reset();
196
197 // Reset the power control properties and control loop histories
198 Dev_PwrCtrl_Reset(&dab);
199
200 #if (CURRENT_CALIBRATION == true)
201 // reset the PWM settings in Standby mode
202 Dev_CurrentSensor_Clr_Offset();
203 // Execute current sensor offset calibration
204 Dev_CurrentSensorOffsetCal();
205 // Checks if the calibration is complete
206 if (Dev_CurrentSensor_Get_CalibrationStatus())
207 #endif
208 {
209 pcInstance->Data.ISecSensorOffset = Dev_CurrentSensor_Get_Offset();
210
211 // Enable current control loop
212 pcInstance->ILoop.Enable = true;
213 pcInstance->ILoop.AgcFactor = dab.ILoop.AgcFactor;
214
215 // Update PWM distribution
216 Dev_PwrCtrl_PWM_Update(&dab);
217
218 // Enable PWM physical output
219 Dev_PwrCtrl_PWM_Enable();
220
221 // Enable power control running bit
222 pcInstance->Status.bits.Running = 1;
223
224 // Next State assigned to STATE_SOFT_START
225 pcInstance->State = PWRCTRL_STATE_SOFT_START;
226 }
227 }
228}

242static void PCS_SOFT_START_handler(POWER_CONTROL_t* pcInstance)
243{
244 // Check for fault event
245 if (pcInstance->Fault.FaultDetected)
246 {
247 // Clear power control enable bit
248 pcInstance->Properties.Enable = 0;
249
250 // State back to STATE_FAULT_DETECTION
251 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
252 }
253
254 // Check if Enable bit has been cleared
255 else if (!pcInstance->Properties.Enable)
256 {
257 // Disable PWM physical output
258 Dev_PwrCtrl_PWM_Disable();
259
260 // Clear power control running bit
261 pcInstance->Status.bits.Running = 0;
262
263 // State back to STATE_STANDBY
264 pcInstance->State = PWRCTRL_STATE_STANDBY;
265 }
266
267 else
268 {
269 // Ramp Up the Voltage, Current and Power reference
270 Dev_PwrCtrl_RampReference(&pcInstance->VRamp);
271 Dev_PwrCtrl_RampReference(&pcInstance->IRamp);
272 Dev_PwrCtrl_RampReference(&pcInstance->PRamp);
273
274 // Check if ramp up is complete
275 if ((pcInstance->VRamp.RampComplete) && (pcInstance->IRamp.RampComplete)
276 && (pcInstance->PRamp.RampComplete))
277 // Next State assigned to STATE_ONLINE
278 pcInstance->State = PWRCTRL_STATE_ONLINE;
279
280 }
281}

293static void PCS_UP_AND_RUNNING_handler(POWER_CONTROL_t* pcInstance)
294{
295 // Check for fault event
296 if (pcInstance->Fault.FaultDetected)
297 {
298 // Clear power control enable bit
299 pcInstance->Properties.Enable = 0;
300
301 // State back to STATE_FAULT_DETECTION
302 pcInstance->State = PWRCTRL_STATE_FAULT_DETECTION;
303 }
304
305 else
306 {
307 // Check if Enable bit has been cleared
308 if (!pcInstance->Properties.Enable)
309 {
310 // Disable PWM physical output
311 Dev_PwrCtrl_PWM_Disable();
312
313 // Clear power control running bit
314 pcInstance->Status.bits.Running = 0;
315
316 // State back to STATE_INITIALIZE
317 pcInstance->State = PWRCTRL_STATE_INITIALIZE;
318 }
319
320 #if (OPEN_LOOP_PBV == true)
321 else if ((pcInstance->Pwm.ControlPeriod & ~(0x7)) != (pcInstance->Pwm.PBVPeriodTarget& ~(0x7)) ||
322 (pcInstance->Pwm.ControlPhase != pcInstance->Pwm.PBVControlPhaseTarget))
323 pcInstance->State = PWRCTRL_STATE_SOFT_START;
324 #endif
325
326 // Check if there is change in power control references
327 else if ((pcInstance->ILoop.Reference != pcInstance->Properties.IReference) ||
328 (pcInstance->VLoop.Reference != pcInstance->Properties.VSecReference) ||
329 (pcInstance->PLoop.Reference != pcInstance->Properties.PwrReference))
330
331 // State back to STATE_SOFT_START
332 pcInstance->State = PWRCTRL_STATE_SOFT_START;
333 }
334}

345static void Dev_PwrCtrl_Reset(POWER_CONTROL_t* pcInstance)
346{
347 //set the period to maximum
348 pcInstance->Pwm.ControlPeriod = MAX_PWM_PERIOD;
349 pcInstance->Pwm.ControlPhase = dab.Pwm.DeadTimeLow;
350 pcInstance->Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
351 pcInstance->Pwm.PBVControlPhaseTarget = dab.Pwm.DeadTimeLow;
352
353 // Reset the power control references
354 pcInstance->Properties.VPriReference = 0;
355 pcInstance->Properties.VSecReference = 0;
356 pcInstance->Properties.IReference = 0;
357 pcInstance->Properties.PwrReference = 0;
358
359 // Initialize current loop reference to 0, to be controlled externally
360 pcInstance->ILoop.Reference = 0;
361 // Initialize power loop reference to 0, to be controlled externally
362 pcInstance->PLoop.Reference = 0;
363 // Initialize voltage loop reference to current secondary voltage
364 pcInstance->VLoop.Reference = 0;
365
366 // Set the AGC to 1
367 pcInstance->VLoop.AgcFactor = 0x7FFF;
368 pcInstance->ILoop.AgcFactor = 0x7FFF;
369 pcInstance->PLoop.AgcFactor = 0x7FFF;
370
371 // Reset Control Loop Histories
372 Dev_PwrCtrl_ResetControlLoopHistories();
373
