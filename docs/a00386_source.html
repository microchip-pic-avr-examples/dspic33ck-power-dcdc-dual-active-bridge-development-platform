114 VMC_2p2z.controlHistory = &VMC_2p2zControlHistory[0]; // Set up pointer to controller history

120 VMC_2p2z.minOutput = DAB_SPVO_COMP_2P2Z_MIN_CLAMP;
121 VMC_2p2z.maxOutput = DAB_SPVO_COMP_2P2Z_MAX_CLAMP;
122
123 VMC_2p2zACoefficients[0] = DAB_SPVO_COMP_2P2Z_COEFF_A1;
124 VMC_2p2zACoefficients[1] = DAB_SPVO_COMP_2P2Z_COEFF_A2;
