63void Dev_PwrCtrl_Initialize(void)
64{
65 // Clear dab references
66 dab.Properties.VPriReference = 0;
67 dab.Properties.VSecReference = 0;
68 dab.Properties.IReference = 0;
69 dab.Properties.PwrReference = 0;
70
71 // Initialize the Control Period and Control Phase during start-up
72 dab.Pwm.ControlPeriod = MAX_PWM_PERIOD;
73 dab.Pwm.DeadTimeHigh = MINIMUM_DEADTIME;
74 dab.Pwm.DeadTimeLow = MINIMUM_DEADTIME;
75 dab.Pwm.PBVPeriodTarget = MAX_PWM_PERIOD;
76
77 // Initialize the DAB to charging state
78 dab.PowerDirection = PWR_CTRL_CHARGING;
79
80 // Initialize Start-Up ramp settings
81 Dev_PwrCtrl_StartUpInitialize();
82
83 // Initialize Power Control Loop
84 Dev_PwrCtrl_ControlLoopInitialize();
85
86 // Disable PWM output by setting the PWM override bits to High
87 Dev_PwrCtrl_PWM_Disable();
88
89 // Enable PWM peripheral
90 PWM_Enable();
91
92 // Update the Period, Duty Cycle and Phases of the PWMs based on
93 // the given Control period and Control Phase
94 Dev_PwrCtrl_PWM_Update(&dab);
95}

165static void Dev_PwrCtrl_StartUpInitialize(void)
166{
167 // Initialize Voltage ramp-up settings
168 dab.VRamp.ptrReference = (uint16_t*)&dab.VLoop.Reference;
169 dab.VRamp.ptrReferenceTarget = &dab.Properties.VSecReference;
170 dab.VRamp.StepSize = 1;
171 dab.VRamp.Delay = 20;
172 dab.VRamp.Counter = 0;
173 dab.VRamp.RampComplete = 0;
174
175 //Initialize Current ramp-up settings
176 dab.IRamp.ptrReference = (uint16_t*)&dab.ILoop.Reference;
177 dab.IRamp.ptrReferenceTarget = (uint16_t*)&dab.Properties.IReference;
178 dab.IRamp.StepSize = 1;
179 dab.IRamp.Delay = 60;
180 dab.IRamp.Counter = 0;
181 dab.IRamp.RampComplete = 0;
182
183 //Initialize Power ramp-up settings
184 dab.PRamp.ptrReference = (uint16_t*)&dab.PLoop.Reference;
185 dab.PRamp.ptrReferenceTarget = &dab.Properties.PwrReference;
186 dab.PRamp.StepSize = 1;
187 dab.PRamp.Delay = 4;
188 dab.PRamp.Counter = 0;
189 dab.PRamp.RampComplete = 0;
190
191#if (OPEN_LOOP_PBV == true)
192 // The PWM Period bits [2:0] needs to be mask when using cascaded PWM setup
193 // (please refer to Section 4.1.3.3 in High Resolution PWM FRM)
194 uint16_t PeriodMask = 0x7;
195
196 // Initialize Voltage ramp-up settings for Period control
197 dab.Pwm.ControlPeriod = dab.Pwm.ControlPeriod & ~(PeriodMask);
198 dab.Pwm.PBVPeriodTarget = dab.Pwm.PBVPeriodTarget & ~(PeriodMask);
199
200 dab.VRamp.ptrReference = &dab.Pwm.ControlPeriod;
201 dab.VRamp.ptrReferenceTarget = &dab.Pwm.PBVPeriodTarget;
202 dab.VRamp.StepSize = 0x7;
203 dab.VRamp.Delay = 0;
204
205 //Initialize Current ramp-up settings for Phase control
206 dab.Pwm.ControlPhase = dab.Pwm.ControlPhase & ~(PeriodMask);
207 dab.Pwm.PBVControlPhaseTarget = dab.Pwm.PBVControlPhaseTarget & ~(PeriodMask);
208
209 dab.IRamp.ptrReference = &dab.Pwm.ControlPhase;
210 dab.IRamp.ptrReferenceTarget = &dab.Pwm.PBVControlPhaseTarget;
211 dab.IRamp.StepSize = 0x7;
212 dab.IRamp.Delay = 0;
213
214#endif
215}
